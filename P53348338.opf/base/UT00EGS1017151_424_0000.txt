༼ཕ༽༄༅། །དྲིལ་བུ་པའི་དབང་གི་བྱ་བ་བསྡུས་པ་བཞུགས་སོ།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། སྐུ་གསུམ་ཡེ་ཤེས་ལྔ་ཡི་དངོས། །མཁའ་འགྲོའི་དབང་ཕྱུག་བཅོམ་ལྡན་འདས། །འཁོར་ལོ་སྡོམ་པ་ལ་བཏུད་དེ། །དབང་གི་བྱ་བ་མདོར་བསྡུས་བཤད། །འཁོར་ལོ་གསུམ་རིམ་རྣལ་འབྱོར་ཉིད། །གང་གིས་ཁམས་གསུམ་སེམས་ཅན་རྣམས། །རྡོ་རྗེ་འཆང་གི་གནས་ཐོབ་པའི། །དབང་བསྐུར་ཆོ་གའི་རིམ་པ་བཤད། །དབང་བསྐུར་དཀྱིལ་འཁོར་སྔོན་འགྲོ་བར། །རྡོ་རྗེ་འཆང་གིས་གསུངས་པ་སྟེ། །འོན་ཀྱང་དཀྱིལ་འཁོར་དེའི་རང་བཞིན། །ཅི་ཡིན་དེ་ནི་འདི་ལྟར་བཤད། །རི་མོར་གནས་པའི་ལས་དང་ནི། །ཐིག་ལ་ཚོན་དབྱེའི་རིམ་པ་བསྟན། །འགྲོ་བ་འདི་དག་རང་བཞིན་གྱིས། །གྲུབ་པའི་དཀྱིལ་འཁོར་གཉིས་མེད་པའོ། །བཅོས་མ་གཉིས་ཀྱི་ངོ་བོ་གང༌། །དེ་ནི་གདུལ་བྱའི་དབང་ལས་འདོད། །མཁས་པས་བསྒྲུབ་བྱ་དེ་མིན་ཏེ། །ཡང་དག་དོན་མཐོང་གྲོལ་ཕྱིར་རོ། །འཁོར་ལོ་གསུམ་གྱི་དཀྱིལ་འཁོར་གྲུབ། །དཀྱིལ་འཁོར་ལྷ་རྣམས་གྲུབ་པའོ། །དེ་ཡི་དབང་བསྐུར་ལ་སོགས་རིམ། །ཅི་ནས་མ་གྲུབ་ཡང་གྲུབ་བྱེད། །དེ་ཡི་རང་བཞིན་མི་ཤེས་རྣམས། །སྣོད་དུ་བྱ་བར་བཞེད་པ་སྟེ། །དབང་བསྐུར་སོགས་རིམ་ཀུན་རྫོབ་ཏུ། །ཐ་སྙད་དུ་ཡང་གྲུབ་པའོ། །སློབ་དཔོན་ལ་སོགས་དབྱེ་བ་ཡིས། །དབང་བསྐུར་རྣམ་པ་བཞིར་བཤད་དེ། །ཆུ་ཡི་དབང་སོགས་དབྱེ་བ་ཡིས། །བཞི་པོ་དེ་ནི་བཅུ་བཞིའོ། །དེའང་ལྔ་སྟེ། དབང་བསྐུར་གྱི་སྔོན་དུ་འགྲོ་བའི་འཐད་པ། གང་དུ་འཇུག་པའི་ཡུལ་དཀྱིལ་འཁོར་གྱི་རབ་ཏུ་དབྱེ་བ། བཅོས་མ་གཉིས་སྤངས་ཏེ་རང་བཞིན་གྱིས་གྲུབ་པའི་དཀྱིལ་འཁོར་ལ་བརྟེན་པའི་ཚུལ། དོན་དམ་པར་སྤྲོས་པ་ལས་གྲོལ་ཡང་ཀུན་རྫོབ་ཏུ་དབང་བསྐུར་བའི་འཐད་པ། དབང་གི་དབྱེ་བ་སྤྱིར་བསྟན་པའོ། །བསྙེན་པ་སྔོན་དུ་སོང་བ་ཡིས། །དང་པོར་བདག་གི་རྣལ་འབྱོར་བྱ། །སྔགས་སོགས་ཀྱིས་མཆོད་གཏོར་མ་སྦྱིན། །བུམ་པའི་ཆོ་ག་ལེགས་བྱས་ལ། །དཔའ་བོ་རྣལ་འབྱོར་མ་རྣམས་མཆོད། །སྤྱི་ཡི་རྣམ་གཞག་སྔོན་འགྲོ་བས། །མཎྜལ་ཕུལ་ལ་བདེན་པ་བརྗོད། །གདོང་གཡོགས་བཅིངས་ལ་མེ་ཏོག་གཏད། །བསྲུང་ཞིང་བྱིན་རླབས་འདོད་པ་དྲི། །བྱང་སེམས་བསྐྱེད་དེ་གསང་བར་གདམས། །ནང་དུ་བཅུག་ལ་གསོལ་བ་གདབ། །དམ་ཚིག་བསྒྲགས་པ་ཐུན་མོང་དང༌། །ཁྱད་པར་གྱི་ནི་སྡོམ་པ་སྦྱིན། །ལེགས་སྦྱངས་དངོས་གཞིའི་ཆོ་ག་བྱ། །དམ་ལ་བཞག་ཅིང་ཡེ་ཤེས་དབབ། །མེ་ཏོག་དོར་ལ་དཀྱིལ་འཁོར་བསྟན། །འཇུག་པ་ཡི་ནི་ཆོ་ག་འོ། །དབང་གི་དོན་དུ་གསོལ་བ་གདབ། །དབང་བསྐུར་རྣམ་པ་བཅུ་བཞི་ཡིས། །དེ་ནས་དབང་ནི་བསྐུར་བར་བྱ། །སྔགས་དང་ཆུ་དང་ཅོད་པཎ་དང༌། །རྡོ་རྗེ་དྲིལ་བུ་མིང་གི་དབང༌། །བརྟུལ་བཞུགས་གཉིས་དང་ལུང་བསྟན་དང༌། །སློབ་དཔོན་དབང་དང་རྗེས་གནང་དང༌། །གསང་བ་ཤེས་རབ་ཡེ་ཤེས་དང༌། །བཞི་པ་ཡང་ནི་དེ་བཞིན་ནོ། །སློབ་དཔོན་ལ་སོགས་བཞི་པོ་ཡང༌། །སྔོན་འགྲོ་དངོས་གཞི་མཇུག་གི་ལས། །གསུམ་གསུམ་གྱིས་ནི་ཡང་དག་བསྡུས། །གཞན་དོན་བྱ་བའི་རིམ་པའོ། །རང་གིས་དམ་ཚིག་གསོ་འདོད་ན། །མདུན་དུ་དཀྱིལ་འཁོར་བསྒྲུབས་ལ་མཆོད། །སློབ་དཔོན་གྱི་ནི་ཆོ་ག་རྣམས། །དཀྱིལ་འཁོར་གཙོ་བོས་མཛད་པར་བསམ། །ཆོ་གའི་གོ་རིམ་སྔར་དང་མཚུངས། །རྗེས་ཀྱི་བྱ་བ་གཞན་དང་འདྲ། །འོན་ཀྱང་གཏོར་མགྲོན་མ་གཏོགས་པ། །གཞན་ནི་གཤེགས་སུ་གསོལ་བ་མེད། །ལུས་ལ་དཀྱིལ་འཁོར་ལ་བརྟེན་ཕྱིར། །དབང་གི་བྱ་བ་མདོར་བསྡུས་པའོ། །ཟབ་མོའི་རྒྱུད་ལ་བརྟེན་ནས་ནི། །རྣལ་འབྱོར་དབང་ཕྱུག་དྲིལ་བུ་པས། །དབང་བསྐུར་ཆོ་གའི་རིམ་པ་མཛད། །དེ་ཡི་མདོར་བསྡུས་བདག་གིས་སྦྱར།
