༼ན༽༄༅། །ལྷ་སའི་བདེ་བར་གཤེགས་པ་རྣམས་ལ་བསྟོད་པ་བཞུགས།། ༄༅། །བླ་མ་དང་འཇམ་པའི་དབྱངས་ལ་ཕྱག་འཚལ་ལོ། །ཟླ་བ་ཁ་བ་ཀུནྡ་གངས་རིའི་མདོག །དྲི་མེད་མུ་ཏིག་རྡོ་ཐལ་དུང་འདྲའི་སྐུ། །འཇིག་རྟེན་དབང་ཕྱུག་དྲི་མས་མ་གོས་པ། །ཐུགས་རྗེ་མངའ་བ་ཁྱོད་ལ་བདག་སྐྱབས་མཆི། །འདིར་སྨྲས་པ། བྱང་ཕྱོགས་ཀྱི་རྒྱུད། ཁ་བ་ཅན་གྱི་ལྗོངས། སྟོན་ཀའི་ནམ་མཁར་རི་བོང་འཛིན་པའི་བསིལ་བའི་འོད་ཟེར་ཤར་བ་ལྟར། ཆེས་ཆེར་དཀར་བའི་གངས་རི་དཔལ་དང་ལྡན་པའི་ཚོགས་ཀྱིས་ཀུན་ནས་བསྐོར་བ། རིའི་སྨན་ལྗོངས་ཁ་བའི་དུམ་བུ་འཁྲིགས་པའི་རྡུལ་གྱིས་གཏམས་པའི་ས་གཞིའི་ངོགས་ལ། པདྨའི་རྩ་ལག་སྐྱེ་དགུའི་སྒྲོན་མ། དཔལ་ཀུན་དུ་འབར་བའི་ཚ་ཟེར་ཕོག་པས་རབ་ཏུ་བཞུ་བའི་ཆུ་ཀླུང༌། བསིལ་ཞིང་དྭངས་ལ་ཡིད་དུ་འོང་བའི་ཅ་ཅོའི་སྒྲ་སྙན་པ་སྒྲོགས་པ་དུ་མ་འབབ་པ། རིམས་ནད་རབ་ཏུ་འཕྲོགས་པའི་སྨན་རྣམ་པ་སྣ་ཚོགས་པའི་རྩ་བ་དང༌། སྡོང་བུ་དང༌། ཤུན་ལྤགས་དང༌། ལོ་འདབ་དང༌། མེ་ཏོག་དང༌། འབྲས་བུ་དུ་མའི་སྐྱེ་གནས། རྡོ་རྗེ་དང༌། ནོར་བུ་དང༌། སྤུག་དང༌། རྒྱལ་པོ་འཁྱིལ་པ་དང༌། ཨནྡ་རྙིལ་དང༌། དངུལ་དང༌། ས་ལེ་སྦྲམ་ལ་སོགས་པ་རིན་པོ་ཆེའི་རིགས་དུ་མའི་འབྱུང་ཁུངས། བདེ་བར་གཤེགས་པ་ཉིད་ཀྱིས་དམ་པའི་ཆོས་བྱང་ཕྱོགས་དང་བྱང་ཕྱོགས་ཀྱི་ཆར་རྒྱས་པར་འགྱུར་རོ། །ཞེས་ལུང་བསྟན་པའི་བོད་ཀྱི་སའི་ཆ་འདིར། སྨོན་ལམ་གྲུབ་པ་བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་ལ་གཅིག་ཏུ་གཞོལ་ཞིང་བསྙེན་བསྲིང་བ་དང་བྲལ་བའི་ཐུགས་རྗེ་ཅན། གཏིང་མཐའ་མེད་པའི་འཁོར་བའི་རྒྱ་མཚོ་ལས་འཕགས་པའི་གང་ཟག་རིན་པོ་ཆེའི་གླིང་དུ་ཕྱིན་པར་བྱེད་པའི་དེད་དཔོན། ལྷ་དང་བཅས་པའི་འཇིག་རྟེན་ཐམས་ཅད་ཀྱི་སྐྱབས་དང་སྐྱོབ་པ་བླ་ན་མེད་པ་སྤྱན་རས་ཀྱིས་གཟིགས་པ་ཙམ་གྱིས་མངོན་པར་མཐོ་བ་དང་ངེས་པར་ལེགས་པའི་རྒྱུར་འགྱུར་བ། རྒྱལ་བ་ཐམས་ཅད་ཀྱི་སྲས་ཀྱི་ཐུ་བོ། བྱང་ཆུབ་སེམས་དཔའ་ཆེན་པོ་འཕགས་པ་འཇིག་རྟེན་དབང་ཕྱུག་ཐུགས་རྗེ་ཆེན་པོའི་ཞབས་ལ་ཕྱག་འཚལ་ཞིང་དུས་ཐམས་ཅད་དུ་གསོལ་བ་འདེབས་སོ། །ཐུབ་པའི་གསུང་རབ་ཆར་སྤྲིན་ལས་སྐྱེས་ཤིང༌། །མཁས་པ་ཚོགས་པ་གངས་རིའི་ཁྲོད་ནས་བྱུང༌། །བློ་གསལ་དྭངས་པའི་ཆུ་གཙང་སྟོང་འདྲེས་པའི། །དཔལ་ལྡན་ཀུན་དགའི་གངྒཱ་རྒྱུན་ཕྱོགས་འབབ། །བདེ་གཤེགས་ཡོན་ཏན་རྒྱ་མཚོར་འཇུག་པ་དེ། །བཅོམ་ལྡན་ལྷ་སའི་སྐྱེད་ཚལ་འཆུ་འདོད་ཀྱང༌། །སྐྱེ་བོའི་ཚོགས་ཀྱི་གཅོང་རོང་ནང་ཞུགས་ཤིང༌། །སློབ་མ་འདུས་པའི་ཆུ་ལོན་དམ་པས་ན། །སྡེ་བཞིའི་ཡུར་བར་རེ་ཞིག་ཆུད་པར་དཀའ། །ཕྱི་ནས་གོ་སྐབས་རྩིག་པ་ལེགས་བསྒྲུབས་ཏེ། །སྣ་ཚོགས་ཐབས་ཀྱི་ཟོ་ཆུན་འཁྲུལ་འཁོར་ལ། །བྱེད་པ་དམ་པའི་སྐྱེས་བུས་མྱུར་བསྐུལ་ནས། །བྱ་བ་བསྒྲུབ་པའི་ཆུ་རྒྱུན་རབ་ཕབ་སྟེ། །དགོས་པ་མཆོག་གི་ལྡུམ་རར་ལེགས་བཅུས་ནས། །ཕྲིན་ལས་མེ་ཏོག་འབྲས་བུ་མྱུར་སྨིན་པར། །རྒྱལ་དང་རྒྱལ་བའི་སྲས་ལ་བདག་གསོལ་འདེབས། །སྒྲ་པ་ང་ཡིན་རྟོག་གེ་པ་ང་སྨྲ་བ་ངན་འཇོམས་ང་འདྲ་མེད། །སྡེབ་སྦྱོར་ངས་ཤེས་སྙན་ངག་ང་ཉིད་མངོན་བརྗོད་འཆད་ལ་འགྲན་མེད་ང༌། །དུས་སྦྱོར་ང་མཁས་ཕྱི་ནང་ཀུན་རིག་རྣམ་དཔྱོད་བློ་གྲོས་མཚུངས་མེད་ང༌། །དེ་འདྲ་གང་ཡིན་ས་སྐྱ་པ་སྟེ་མཁས་པ་གཞན་དག་གཟུགས་བརྙན་ཡིན། །ལྷ་སའི་བདེ་བར་གཤེགས་པ་ཐམས་ཅད་ལ། །རབ་ཏུ་བསྔགས་པའི་ཚིག་འགའ་གྱིས་ཤིག་ཅེས། །བློ་གྲོས་གསལ་ལྡན་བླ་མ་ལ་གུས་ཡིད། །ཆོས་སྨྲ་ཆོས་ཀྱི་དཔལ་གྱིས་བསྐུལ་ནས་བྲིས།།

།། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
