༼ན༽༄༅། །ཡང་ན་མན་ངག་བསྡུས་པ་གཅིག་བཞུགས།། ༄༅། །ན་མོ་གུ་ར་ཝེ། གསལ་བྱེད་ཉི་མའི་དཀྱིལ་འཁོར་འདི་ལ། རྩ་བའི་གཞུང༌། མཐའ་འགྲོལ་གྱི་ཡིག་སྣ། སྤོགས་པའི་ཆོ་ག སྤྱིའི་གདམས་ངག་གོ །མཐའ་འགྲོལ་གྱི་ལག་ལེན་ལ། འདོད་རྒྱལ་ཡབ་ཡུམ་གྱི་སྒོ་ནས་དབང་དྲག་གཉིས་སུ་བསྒྲུབ་པ། ཡབ་རྐྱང་པའི་སྒོ་ནས་བསྒྲུབ་པ། ཡུམ་རྐྱང་པའི་སྒོ་ནས་བསྒྲུབ་པའོ། །སྤོགས་པའི་ཆོ་ག་ལ་གསལ་བྱེད་ཉི་ཟླའི་འཁོར་ལོའི་སྤོགས་ལ། དབྱངས་དང་གསལ་བྱེད་ཀྱིས་དཀྱིལ་འཁོར་དུ་སྦྱིན་སྲེག་བྱ་བའོ། །ལོ་རྒྱུས། རབ་གནས། རང་བསྲུང༌། ལས་བཞིའི་གདམས་པའོ། །རྩ་བའི་འཁོར་ལོ་ཡི་གེ་གཡོན་བསྐོར་དུ་བྲིས་པ་འོག གཡས་སྐོར་དུ་བྲིས་པ་སྟེང༌། ཡི་གེ་གཅིག་ཀྱང་མ་བསྒུལ་བར་ཁ་སྦྱོར་དུ་བྱ་སྟེ། ལྟེ་བ་དང་བར་སྐབས་སྤེལ་ཚིག་ཐོབ་པ་རྣམས་སུ་མིང་སྤེལ་བྲི། ཕོ་འཁོར་མོ་འཁོར་ཉི་ཟླ་དང་བཅས་པ་རྣམས་དར་དམར་པོ་ལ་སྔགས་རིང་བྲིས་པས་གྲིལ། ཕྱི་ཕྲུ་མ་དམར་པོར་བསྒྱུར་བས་གྲིལ། སྣག་ཚ་ཚོས་ཀྱི་གསར་པ་ལ་གཞན་རྣམས་དང་ཅུང་ཟད་བསྲེ། གཞི་དར་དམར་ལ་ས་མན་རྡོ་མན་ཅུང་ཟད་བྱུགས། སྤྱིན་ཆུ་བསྐུས་ལ་ཨུར། རབ་གནས་ཀྱི་དཀྱིལ་འཁོར་ཟླ་གམ་སྒོ་ནུབ་ཏུ་བསྟན་པའི་ནང་པད་འདབ་ལ་སོགས་པ་བྲི། གསང་ནས་བསྒྲུབས་པས་རབ་གནས་ཀྱིས་གྲུབ་པའང་བྱུང་སྐད། མིང་སྤེལ་གྱི་བྲི་ས་སྟོང་པར་བཞག་ནས་དར་གཞན་མ་བྲིས་པ་བག་གིས་སྦྱར། ཕྱིས་སྤོ་ནའང་དེ་བཀོག་ནས་གཞན་སྦྱར་བས་ཆོག་གསུང༌། སྔགས་དབང་ལ། ཀྵ་ཡམ་དང༌། ཀྵ་བྷ་ཀུ་རུ་བྲི་ཞིང་བཟླས། ཞི་བ་ལ་ཤཱནྟིཾ་ཀུ་རུ། རྒྱས་པ་ལ་པུཥྚིཾ་ཀུ་རུ། དྲག་པོ་ལ་མཱ་ར་ཡ་ཀུ་རུ་བྲི་བ་དང་བཟླས་སོ། །ཁྲོ་བོ་འདོད་པའི་རྒྱལ་པོའི་མན་ངག་གི་གནད་ཐམས་ཅད་ཕྱོགས་གཅིག་ཏུ་བསྡེབས་པ་ཨི་ཐི།། །།གཞན་ལ་མེད་པའི་མན་ངག་ཁྱད་པར་ཅན། །བདག་གིས་འབད་དེ་བྲིས་པའི་དགེ་བ་ཡིས། །མྱུར་དུ་འདོད་པའི་རྒྱལ་པོ་འགྲུབ་གྱུར་ནས། །འགྲོ་བའི་དབུལ་བ་སེལ་བའི་མཐུ་ལྡན་ཤོག
