༼ཕ༽༄༅། །ཕྱག་དྲུག་སྒྲོལ་མའི་སྒྲུབ་ཐབས་བཞུགས།། ༄༅། །ཕྱག་དྲུག་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། རྗེ་བཙུན་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །འགྲོ་བ་སྡུག་བསྔལ་ཆུ་བོ་ལས། །སྒྲོལ་མཛད་སྒྲོལ་མར་ཕྱག་འཚལ་ནས། །སྒྲོལ་མའི་སྒྲུབ་ཐབས་སྒྱུ་འཕྲུལ་གྱི། །རྒྱུད་ཀྱི་རྗེས་སུ་འབྲང་སྟེ་བཤད། །དེ་ལ་ཐོག་མར་སྔགས་པས་ཕྱི་དང་ནང་གི་བདག་ཉིད་ཀྱི་ཁྲུས་ཀྱིས་ལུས་དང་སེམས་དག་པར་བྱས་ནས། སྐྱབས་སུ་འགྲོ་བ་དང་སེམས་བསྐྱེད་པ་སྔོན་དུ་སོང་ནས་ཡན་ལག་བདུན་པ་དང་ཚད་མེད་པ་བཞིས་བསོད་ནམས་ཀྱི་ཚོགས་བསགས་ལ། ཨོཾ་ཤཱུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོ྅་ཧཾ་གི་སྔགས་ཀྱིས་ཆོས་ཐམས་ཅད་མི་དམིགས་པའི་སྟོང་པར་བསྒོམས་ལ། སྟོང་པའི་ངང་ལས་པྃ་ལས་པདྨ་ཨ་ལས་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་བསྒོམ་པར་བྱ་བ་ནི། ཏཱྃ་གི་ཡི་གེ་ལས་གྲུབ་པ། །ལྷ་མོ་སྒྲོལ་མ་རྫུ་འཕྲུལ་ཆེ། །འཇོ་སྒེག་ཆ་ལུགས་ཀྱིས་མཛེས་པ། །ཞལ་རས་གསུམ་པ་སྤྱན་གསུམ་པ། །ཕྱག་དྲུག་སེམས་དཔའི་སྐྱིལ་མོ་ཀྲུང༌། །བཞད་པ་མརྒད་མདོག་དང་ལྡན། །དང་པོ་མི་འཇིགས་སྦྱིན་པའི་ཕྱག །གཉིས་པ་ན་ནི་རྡོ་རྗེ་བསྣམས། །གསུམ་པ་ཡིས་ནི་མདའ་ཡང་གསོར། །གཡོན་པས་སྡིགས་པའི་མཛུབ་མོ་སྟེ། །གཉིས་པ་ན་ནི་ཨུཏྤལ་ཉིད། །གསུམ་པ་ཡིས་ནི་གཞུ་ཡང་འཛིན། །འོད་ཟེར་དམར་པོ་འབར་བ་ཅན། །སངས་རྒྱས་འཕོ་བས་རྒྱས་མཛད་པ། །དགའ་མཛད་སེམས་ཅན་ཀུན་གྱི་མ། །རྟག་པར་རྒྱུན་དུ་བསྒོམ་པར་བྱ། །སེམས་ཅན་ཐམས་ཅད་དབང་གྱུར་ཞིང༌། །དགའ་བར་བྱེད་པའི་ཡེ་ཤེས་རྡོ་རྗེ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ཏོ། །དེའི་གནས་གསུམ་དུ། ཨོཾ་ཨཱཿཧཱུྃ་གསུམ་བཀོད་ནས་ཡེ་ཤེས་པ་དང་རོ་གཅིག་པར་བྱས་ལ་དབང་གི་ལྷ་སྤྱན་དྲངས་ནས་གསོལ་བ་བཏབ་པས་དབང་བསྐུར་ནས་རིགས་ཀྱི་བདག་པོ་དོན་ཡོད་གྲུབ་པའི་རྒྱས་གདབ་བོ། །དེ་ནས་ལག་པ་གཉིས་ཐལ་མོ་སྙིང་གར་སྦྱར་ལ་མཛུབ་མོ་གཉིས་ཀྱི་རྩེ་མོས་གུང་མོའི་ཚིགས་དང་པོར་རེག་ནས་མཐེ་བོང་གཉིས་སྦྱར་ཏེ་དབུས་སུ་བཞག་པ་ཨུཏྤ་ལའི་ཕྱག་རྒྱ་བཅས་པས།

དཔྲལ་བ། མགྲིན་པ། སྙིང་གར་གཏུག་ཅིང༌། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཏཱ་ར་ཀཱ་ཡ་ཨ་དྷི་ཏིཥྛ་ཨོཾ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཏཱ་ར་ཝཱཀ་ཨ་དྷི་ཏིཥྛ་ཨཱཿ ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཏཱ་ར་ཙིཏྟ་ཨ་དྷི་ཏིཥྛ་ཧཱུྃ། ཞེས་རིམ་པས་བརྗོད་ཅིང་བྱིན་གྱིས་བརླབ་བོ། །དེ་ནས་ལྷའི་སྐུ་སྣང་ལ་རང་བཞིན་མེད་པ་ལ་སེམས་གཏད་དེ་མཉམ་པར་གཞག་གོ །དེ་ནས་སྔགས་བཟླ་བར་འདོད་ན། སྙིང་གར་ཟླ་བའི་སྟེང་དུ་ཏཱྃ་གི་མཐའ་མར་སྔགས་ཀྱི་ཕྲེང་བས་བསྐོར་བའི་འོད་ཀྱིས་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བྱིན་རླབས་སྤྱན་དྲངས་ཤིང་རང་ལ་བསྟིམ། རང་དང་གཞན་གྱི་དོན་གཉིས་བསྒྲུབ་པར་བསམ་ཞིང༌། བཟླ་བར་བྱ་བའི་སྔགས། ཕན་ཡོན་དང་བཅས་པ་ནི། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། དེ་ནས་འདི་ནི་གསུངས་ཙམ་གྱིས། །སངས་རྒྱས་ཀུན་དང་སྲས་རྣམས་ཀྱིས། །སྤྱན་ཕྱེ་རྣམ་པར་གདངས་ནས་སུ། །ཡེ་ཤེས་རྡོ་རྗེ་ལྡན་པ་དྲན། །ལྷ་མོ་བཟླས་བརྗོད་བྱས་ཙམ་གྱིས། །སྐྱེ་དགུ་རྣམས་ནི་མངག་གཞུག་འགྱུར། །དེ་ནས་དཀར་གཏོར་ཨོཾ་ཨཱཿཧཱུྃ་གིས་ཡེ་ཤེས་ཀྱི་བདུད་རྩིར་བྱིན་གྱིས་བརླབས་ལ། རང་གིས་བསྒོམས་པ་ལྟ་བུའི་བཅོམ་ལྡན་འདས་མ་མདུན་དུ་སྤྱན་དྲངས་ལ་སྔགས་སྔ་མ་ཉིད་ཀྱི་དབུལ་ལོ། །སླར་ཡང་མཆོད་ཅིང་བསྟོད་ལ་འདོད་པའི་དོན་ལ་གསོལ་བ་བཏབ་ནས། ཨོཾ་བཛྲ་མུཿས་གཤེགས་སུ་གསོལ། འདིའི་སྲུང་བའི་འཁོར་ལོ་དང༌། ལས་ཀྱི་ཚོགས་མང་པོ་ནི་སྒྲོལ་མའི་སྒྲུབ་པའི་ཐབས་ཀྱི་ཡན་ལག་གཞན་ལས་འབྱུང་བ་ལྟར་བསྒྲུབས་པས་འགྲུབ་པ་ཉིད་དོ། །དེ་ལྟར་རྒྱུད་ཀྱི་རིམ་པ་བཞིན། །སྒྲོལ་མའི་སྒྲུབ་ཐབས་བཤད་པ་ཡིས། །འགྲོ་བ་སྲིད་པའི་རྒྱ་མཚོ་ལས། །བརྒལ་ནས་བདེ་བ་སྦྱིན་པར་ཤོག །ཕྱག་དྲུག་སྒྲོལ་མའི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། འཕགས་པས་ས་ཕོ་སྟག་གི་ལོ་དབྱར་ཟླ་ར་བའི་ཚེས་བརྒྱད་ལ་དཔལ་ས་སྐྱར་སྦྱར་བ་འདིའི་ཡི་གེ་པ་ནི་ངག་གི་དབང་ཕྱུག་གྲགས་པའོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
