༼བ༽༄༅། །ཆོས་རྗེ་པ་བདེ་བར་གཤེགས་དུས་དབུས་གཙང་གི་དགེ་བའི་བཤེས་གཉེན་རྣམས་ལ་སྤྲིང་བ་བཞུགས།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། དུས་གསུམ་བདེ་བར་གཤེགས་པ་ཆོས་ཀྱི་སྐུ། །དམ་ཆོས་འཆི་མེད་བདུད་རྩིའི་གཏེར་གྱུར་པ། །བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་བསྐྱེད་པའི་ཡུམ། །མཚུངས་མེད་བླ་མའི་ཞབས་ལ་ཕྱག་འཚལ་ལོ། །བདེ་ལེགས་ཐམས་ཅད་ཀྱི་འབྱུང་གནས་དཔལ་ས་སྐྱིའི་གཙུག་ལག་ཁང་ན་བཞུགས་པའི་དགེ་བའི་བཤེས་གཉེན་རྣམས་ཀྱི་དབུ་མཛད། བླ་མ་ཆོས་རྗེའི་ཞབས་རྡུལ་སྤྱི་བོས་ལེན་པ་དབུས་གཙང་ན་བཞུགས་པའི་བུ་སློབ་རྣམས་ཀྱི་ཕྱག་ཏུ་སྤྲིང་བ། བླ་མ་ཆོས་ཀྱི་རྗེ་དམིགས་པ་མེད་པའི་ཐུགས་རྗེ་ཆེན་པོས་སེམས་ཅན་ཐམས་ཅད་ཡོངས་སུ་སྐྱོང་བར་མཛད་ཅིང༌། རྨད་དུ་བྱུང་བའི་མཁྱེན་རབ་ཕུན་སུམ་ཚོགས་པས་ཤེས་བྱའི་དཀྱིལ་འཁོར་མ་ལུས་པ་ཐུགས་སུ་ཆུད་པའི་བདག་ཉིད་ཆེན་པོ་དེས། རྣམ་པར་འཕྲུལ་པ་བསམ་གྱིས་མི་ཁྱབ་པས་གདུལ་བྱའི་སེམས་ཅན་དུ་མ་ཅིག་གི་དོན་རྒྱ་ཆེན་པོ་མཛད་ནས། སླར་ཡང་འཇིག་རྟེན་གྱི་ཁམས་གཞན་དུ་འགྲོ་བའི་དོན་མཛད་པའི་དུས་ལ་བབ་པར་གཟིགས་པ་དང༌། རྟག་པར་འཛིན་པའི་གདུལ་བྱ་འགའ་ཞིག་ལ་མི་རྟག་པའི་མངོན་པར་རྟོགས་པ་བསྐྱེད་པའི་ཕྱིར་དང༌། ལེ་ལོ་ཅན་རྣམས་ཡང་དག་པའི་ལམ་ལ་ནན་ཏན་གྱིས་བསྒྲུབ་ཏུ་གཞུག་པའི་ཕྱིར། ས་གཡོ་བ་དང༌། སྒྲ་གྲགས་པ་དང༌། ནམ་མཁའ་འོད་ཀྱིས་གང་བ་ལ་སོགས་པའི་ཆོ་འཕྲུལ་བསམ་གྱིས་མི་ཁྱབ་པ་བསྟན་པས། སྐྱེ་བོའི་ཚོགས་ཐམས་ཅད་ཡང་དག་པར་སྡུད་པར་མཛད་ནས། ཁོ་བོ་ཡང་སྐྱེ་བ་གསུམ་པའི་ལུས་ལ་བརྟེན་ནས། དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་དྲི་མ་མེད་པའི་དཔལ་ཞེས་བྱ་བར་མངོན་པར་འཚང་རྒྱ་བར་འགྱུར་རོ་ཞེས། རྗེ་བླ་མ་མངོན་སུམ་དུ་བྱོན་ནས་ལུང་བསྟན་པས། ང་ལ་རྩེ་གཅིག་ཏུ་གསོལ་བ་འདེབས་པ་ཁྱེད་ཐམས་ཅད་ཀྱང༌། དུས་ཐམས་ཅད་དུ་གཟུང་ཞེས་ཞལ་གྱིས་བཞེས་ནས།

ཕག་གི་ལོ་སྨལ་པོའི་ཟླ་བའི་ཡར་གྱི་ངོའི་ཚེས་བཅུ་བཞི་ལ་ལྷའི་ཚོགས་བསམ་གྱིས་མི་ཁྱབ་པའི་འཇིག་རྟེན་ཐམས་ཅད་དང་ཐུན་མོང་དུ་བསྟན་ནས། ཡོངས་སུ་མྱ་ངན་ལས་འདས་པའི་ཚུལ་བསྟན་ཏོ། །དེ་ལ་ཤེས་ལྡན་དག་གིས་མྱ་ངན་དང༌། ཡིད་མི་བདེ་བ་ལ་སོགས་པ་འཇིག་རྟེན་པ་ཐ་མལ་པའི་རྟོག་པ་ཐམས་ཅད་སྤངས་ནས། བསམ་པ་རྣམ་པར་དག་པས་ཆོས་ཀྱི་རྗེ་ཆེན་པོ་ཉིད་ལ་གསོལ་བ་བཏབ་ན། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་སྒྲིབ་པ་མེད་པའི་ཡེ་ཤེས་དང་ལྡན་པའི་ཕྱིར། བསླུ་བ་མེད་པའི་ཡེ་ཤེས་གཟིགས་པ་འཇུག་པར་འགྱུར་ཏེ། སེམས་ཅན་གྱི་དོན་ལ་བཏང་སྙོམས་སུ་འགྱུར་བ་ནམ་དུ་ཡང་མི་མངའ་བའི་ཕྱིར་རོ། །བདག་ཉིད་ཆེན་པོ་སྐུ་གཤེགས་པའི་རྗེས་སུ་ཡང༌། ཆོས་རྗེ་ཉིད་ཀྱི་ཐུགས་རྗེའི་མཐུས་ངེད་ཐམས་ཅད་ཁམས་བདེ་བར་ཡོད། རྒྱལ་བུ་མོང་གོ་ཏ་ཡང་བྱོན། ཐུགས་ལ་འདོགས་པར་ཡོད། མོང་གོ་གན་རྒྱལ་པོར་མངའ་གསོལ་ཕྱོགས་ཐམས་ཅད་དུ་འཇའ་ས་བཟང་པོ་བསྒྲགས། རྒྱལ་ཁམས་ཐམས་ཅད་ཀྱང་ཤིན་ཏུ་བདེ། བྱེ་བྲག་ཏུ་བནྡེ་ལ་དམག་ཁྲལ་ཤོ་དང༌། གན་མཛོད་དུ་ནོར་འདབ་མི་དགོས། བནྡེའི་ཁང་པ་དང་ལྷ་ཁང་དུ་གསེར་ཡིག་པ་འབབ་ར་མེད། འུ་ལག་སྦྱིན་ར་མེད། སོ་སོའི་ཆོས་ལུགས་བཞིན་དུ་གནམ་མཆོད། ངེད་ཐམས་ཅད་ལ་སྨོན་ལམ་ཐོབ། བནྡེ་ཐམས་ཅད་ཀྱི་ཇི་ལྟར་བྱ་བའི་མགོ་ལུང་དེ། ས་སྐྱ་པ་ཤེས་སུ་ཆུག་བྱ་བའི་འཇའ་ས་བཟང་པོ་གནང༌། དེ་ཕྱོགས་མཐའ་དག་ཏུ་བསྒྲགས་པ་དང༌། གསེར་ཡིག་པ་རྣམས་ཀྱི་བོད་སོ་སོའི་མི་རྩི་བ་དང༌། རྒྱལ་ཁམས་གཏན་ལ་འབེབས་པའི་ཆེད་དུ། ཁྱེད་ཀྱི་བནྡེ་ཅིག་ཀྱང་ཐོང་ཟེར་ནས། དེའི་དོན་དུ་དགེ་བཤེས་རྡོ་རྗེ་འབྲུག་དང༌། དགེ་བཤེས་སུམ་བུ་དཔོན་གཡོག་རྣམས་བཏང་བ་ཡིན། བླའོ་ཞིབ་ཏུ་ཁོང་རྣམས་ལ་དྲིས། དེ་ན་ཡར་གཏམ་སྙན་མི་སྙན་སྣ་ཚོགས་ཡོད་ཀྱང་སྐྲག་མི་དགོས། ཕྱོགས་ཀུན་ནས་ལེགས་པར་འོང༌། ཁྱེད་ཐམས་ཅད་ཀྱིས་གོ་ཆ་ཆེན་པོ་གོན་ལ། བླ་མའི་ཞབས་ཏོག་ལ་འབད་པ་སྐྱེད།

ཆུ་ཕོ་བྱི་བའི་ལོ་དཔྱིད་ཟླ་འབྲིང་པོའི་ཚེས་ལྔའི་ཡི་གེ དཔལ་ལིང་ཆུ་རྩིར་ཁབ་ཀྱི་གཙུག་ལག་ཁང་དུ་བྲིས།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
