༼ཉ༽༄། །བསྐྱེད་རིམ་བཟླས་བསྒོམ་ལ་བརྟེན་ནས་མཁའ་སྤྱོད་སྒྲུབ་པ་བཞུགས། །རྡོ་རྗེ་རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་ལོ། །བསྐྱེད་པའི་རིམ་པའི་ཆ་བཟླས་བསྒོམ་གཉིས་ལ་བརྟེན་ནས་མཁའ་སྤྱོད་ཀྱི་དངོས་གྲུབ་སྒྲུབ་པ་ནི། ཡུམ་རྡོ་རྗེ་ཕག་མོ་ལ་བརྟེན་ཏེ། དེའང་བླ་མས་ལྷན་ཅིག་སྐྱེས་པའི་ཕག་མོ་བསྟན་ཅིང་ངོས་ཟིན་པ་ལ་མཆོད་ཅིང་བླ་མས་ཡིད་དམ་དུ་བསྟན་ནས། སློབ་མས་ཀྱང་ཚེས་བཅུ་རྣམ་པ་གཉིས་ལ་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་རྟེན་ལ་མཆོད་ཅིང༌། བཟླས་བསྒོམ་ཡུན་རིང་བྱས་ཤིང་སྒྲུབ་པ་ལ། རྣལ་འབྱོར་པ་དམ་ཚིག་དང་ལྡན་ཞིང་བསོད་ནམས་རབ་ཡིན་ན། རྣལ་འབྱོར་མའི་གཟུགས་ལ་བརྟེན་ཏེ། རྣལ་འབྱོར་པ་ལ་མཁའ་སྤྱོད་ལ་སོགས་པར་བྱིན་གྱིས་རློབ་པར་མཛད་དེ། དེའི་རྟགས་ནི་མི་མོ་གང་ཞིག་ཤ་དང་མིག་སྐྲ་ལ་སོགས་པ་དམར་སེར་དུ་འདུག་ཅིང༌། དཔྲལ་བ་ན་ཤའི་རི་མོ་དགའ་བ་འཁྱིལ་པ་ལྟ་བུ་ཡོད་པ་དེ་ལྟ་བུ་དང་ཕྲད་ན། དེ་ནི་རྣལ་འབྱོར་མ་ཡིན་པའི་རྟགས་ཡིན་ཏེ། དེ་ལ་གཡོན་ནས་བསྐོར་བ་གསུམ་ལ་སོགས་པའི་སྒོ་ནས་མཆོད་པས་མཉེས་པར་བྱས་ཤིང༌། གསོལ་བ་བཏབ་ན་ཚེ་འདི་ཉིད་ལ་མཁའ་སྤྱོད་དུ་འཁྱེར་རོ། །གལ་ཏེ་དེ་ལྟ་བུའི་ཚུལ་གྱིས་གྲུབ་པར་མ་གྱུར་ན། རྣལ་འབྱོར་པ་བསོད་ནམས་འབྲིང་པོ་ཡིན་ཏེ། དེས་ནི་སིནྡྷཱུར་བསྒྲུབ་པའི་ཚུལ་གྱིས་བསྙེན་པ་འབུམ་ཕྲག་བཞི་བསྐྱལ་ལ། དེ་ནས་སིནྡྷཱུ་རའི་སྒྲུབ་པ་བརྩམས་ཏེ། ལངྐ་ལི་ཡ་ཞེས་བྱ་བའི་ཤིང་སྦུ་གུ་ཅན་ཞིག་ཡོད་དེ། དེ་སོར་བཞིའམ། ཡང་ན་བརྒྱད་པ་ཅིག་ཚེས་བཅུ་གང་རུང་ཞིག་ལ་བླངས་ཏེ། རྣལ་འབྱོར་མ་ལ་མཆོད་པ་བྱས་ཤིང༌། དེ་ལག་ཏུ་ཐོགས་ལ་སྟོང་རྩ་བརྒྱད་བཟླས་སོ། །དེ་ནས་ཡར་གྱི་བཅུ་གཅིག་ལ་སིནྡྷཱུ་རའི་ཕྱེ་མ་བཙག་ཅིང་གསེར་གྱི་སྨྱུ་གུས། ཆོས་འབྱུང་དང་སྔགས་དང་དགའ་བ་འཁྱིལ་པ་བྲི་སྟེ། རྟེན་དེ་ལ་མཆོད་ཅིང་བསྟོད་ལ་བཟླས་པ་བྱ། གསོལ་བ་བཏབ་སྟེ་སིནྡྷཱུ་ར་བསྡུས་ལ་སྔགས་བཟླས་པའི་ཤིང་གི་སྦུབས་བཀང་སྟེ།

འོག་གི་བུ་ག་ལ། ལ་ཅའམ་སྤྲ་ཚིལ་གྱིས་བཀག་ལ། སྒྲུབ་པའི་གྲོགས་མཆོག་སུམྦྷ་ནི་ཟློས་པ་ཅིག་དང་དུར་ཁྲོད་དུ་ཕྱིན་ལ་སིནྡྷཱུ་ར་དང་བཅས་པའི་ཤིང་དེ་དུར་ཁྲོད་ཀྱི་ས་བརྐོས་པའི་ནང་དུ་བཅུག་སྟེ། དེའི་སྟེང་དུ་བདག་ཉིད་བསྡད་ལ་བཟླས་པ་ཅི་ནུས་བྱས་ལ་ས་ཕྱོགས་དེ་བསྲུང་བ་བྱས་ལ་བཞག དེ་ནས་བདག་ཉིད་བསམ་གཏན་གྱི་ཁང་བུར་ལོག་ལ་ཇི་ལྟ་བ་བཞིན་བཟླས་སོ། །དེ་ནས་ཚེས་བཅུ་གཅིག་ནས་བཟུང་ནས་བཅོ་ལྔའི་བར་དུ་ནི། ནམ་སུམ་ཆ་དུར་ཁྲོད་དུ་ཕྱིན་ལ་བཟླས་སོ། །དེ་ནས་ཚེས་བཅུ་དྲུག་ནས་ཉི་ཤུའི་བར་དུ་ནི་ནམ་ཕྱེད་ཕྱེད་བཟླས་སོ། །དེ་ནས་ཉི་ཤུ་གཅིག་ནས་བཞིའི་བར་དུ་ནི་སུམ་གཉིས་སུ་བཟླས་སོ། །དེ་ནས་ཉི་ཤུ་ལྔའི་ཚེ་ནི་མཚན་ཐོག་ཐག་ཏུ་བཟླས་ལ། སྐྱ་རེངས་དང་པོ་ཤར་བ་དང༌། སིནྡྷཱུ་ར་བཏོན་ལ་བསམ་གཏན་གྱི་ཁང་པར་སོང་སྟེ། ཡུམ་རྡོ་རྗེ་ཕག་མོ་ལ་མཆོད་ཅིང་བསྟོད། དེ་ལ་གསོལ་བ་བཏབ་ལ། དེ་ལ་ནང་མ་རེ་རེ་ཞིང་བཟླས་སྒོམ་དང་མཆོད་བསྟོད་ལ་སོགས་པ་སྒྲུབ་པ་ཇི་ལྟ་བ་བཞིན་བྱེད་ཅིང༌། བདག་ཉིད་ཀྱི་དཔྲལ་བར་སིནྡྷཱུ་རས་དགའ་བ་འཁྱིལ་པ་བྲི་ཞིང༌། བདག་ཉིད་ས་ལ་འདུག་པའི་བོང་ཚོད་དང་མཉམ་པའི་སྦའི་དབྱུག་པ་ཅིག་ལ་སུམྦྷའི་སྔགས་མང་དུ་བཟླས་པ་ཅིག་ལ། རྣལ་འབྱོར་གྱི་དབྱུག་པར་བྱིན་གྱིས་བརླབས་ལ། དེ་ཐོགས་ནས་ནང་མ་རེ་རེ་ཞིང་སིནྡྷཱུ་རའི་རི་མོ་དཔྲལ་བར་དགའ་བ་འཁྱིལ་པ་བྲི་ཞིང༌། ཡུལ་ཐམས་ཅད་དུ་འཁྱམ་པར་བྱའོ། །འཁྱམས་པ་ལས་ཡུམ་རྡོ་རྗེ་ཕག་མོ་དང་ཕྲད་པར་འགྱུར་ཞིང༌། མི་མོ་ལ་སོགས་པ་གང་དང་ཕྲད་པའི་དུས་སུ་དེའི་རྟགས་ནི་སིནྡྷཱུ་རའི་རི་མོ་དེའི་དཔྲལ་བར་འཕོས་ན། རྡོ་རྗེ་རྣལ་འབྱོར་མ་ཡིན་པས་དེ་ལ་མཆོད་ཅིང་བསྙེན་བཀུར་བྱས་ཏེ་གསོལ་བ་བཏབ་པས་མཁའ་སྤྱོད་ལ་སོགས་པའི་དངོས་གྲུབ་སྟེར་བར་འགྱུར་རོ། །སྒྲུབ་པ་དེ་ལྟ་བུས་ཀྱང་ཚེ་འདི་ཉིད་ལ་གྲུབ་པར་མ་གྱུར་ན། རྣལ་འབྱོར་པ་བསོད་ནམས་ཐ་མ་ཡིན་ཏེ།

བར་མ་དོར་ནུས་པ་བསྐྱེད་ན་བར་མ་དོར་ཡུམ་རྡོ་རྗེ་ཕག་མོའི་ཞལ་མཐོང་ཞིང་མཁའ་སྤྱོད་ལ་སོགས་པའི་དངོས་གྲུབ་སྟེར་བར་འགྱུར་རོ། །བླ་མའི་ཞལ་གྱི་གདམས་པ་ལགས་སོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
