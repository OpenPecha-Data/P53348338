༼པ༽༄༅། །བླ་མ་བརྒྱུད་པའི་མཚན་འབུམ་སྔགས་ཅན་བཞུགས།། ༄༅། །བཛྲ་དྷཱ་ར་བཛྲ་གརྦྷ་དཔལ། །བཛྲ་ཌཱ་ཀ་ཙེ་ལུ་དུས་ཞབས་གཉིས། །དཔེ་མེད་བསྲུང་བ་ལེགས་པའི་སྲས་དང་ནི། །ཆོས་ཀྱི་འབྱུང་གནས་ཆོས་ཀྱི་བཤེས་གཉེན་དང༌། །རྣམ་པར་གྲགས་པའི་ལྷ་དང་ཤཱཀྱ་ཤྲཱི། །བཟང་པོའི་ཞབས་དང་ཆོས་རྗེ་ཆོས་ཀྱི་མགོན། །མཚུངས་མེད་བླ་མའི་ཚོགས་ལ་བདག་སྐྱབས་མཆི། །དུས་རྣམས་ཀུན་ཏུ་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། །ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿནཻ་རཱཏྨ་ཧཱུྃ། ཨོཾ་ཨཱཿབིརྺ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿ ཀཧྞ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཌཱ་མ་རུ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཨ་བ་དྷཱུ་ཏི་ཧཱུྃ། ཨོཾ་ཨཱཿག་ཡ་དྷ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿདྭེ་ཥ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿལ་ལི་ཏ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཔུཎྱ་ཨ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟཱ་བྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱནནྡ་དྷཱ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་ཌཱཀྐི་ནཱི་ཧཱུྃ། ཨོཾ་ཨཱཿབིརྺ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཌོམྦི་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཨ་ལ་ལ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབ་ན་སི་པ་ཧཱུྃ། ཨོཾ་ཨཱཿགརྦྷ་རི་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཤྲཱི་པནྡ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿདུརྫ་ཡ་ཙནྡྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབཱི་ར་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་དྭེ་ཥ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿལི་ལ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཔུཎྱ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟི་དྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་དྷ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་ཌཱ་ཀི་ནཱི་ཧཱུྃ། ཨོཾ་ཨཱཿབི་ལ་སྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཧཱ་ཨ་ནཾ་ག་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཔདྨ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཨིནྡྲ་བྷཱུ་ཏི་ཧཱུྃ། ཨོཾ་ཨཱཿལཀྵྨཱི་པཱ་ད་ཧཱུྃ། ཨོཾ་ཨཱཿཀཧྞ་བ་ཧཱུྃ། ཨོཾ་ཨཱཿཤྲཱི་དྷ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿག་ཡ་དྷ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ།

ཨོཾ་ཨཱཿདྭེ་ཥ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿལི་ལ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཔུཎྱཱ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟི་དྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་པཱ་ཎི་ཧཱུྃ། ཨོཾ་ཨཱཿས་ར་ཧ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿནཱ་གཱརྫུན་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཤ་བ་རི་པ་ཧཱུྃ། ཨོཾ་ཨཱཿལཱུ་ཧི་པ་ཧཱུྃ། ཨོཾ་ཨཱཿདཱ་རི་ཀ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་གྷཎྜེ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཀུརྨ་པཱ་ད་ཧཱུྃ། ཨོཾ་ཨཱཿཤྲཱི་ཛྭཱ་ལ་དྷ་ར་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཀྲྀཥྞ་ཙརྻ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿགུ་ཧྱ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿབི་ཛ་ཡ་པཱ་ད་ཧཱུྃ། ཨོཾ་ཨཱཿཏི་ལོ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿནཱ་རོ་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཨ་བྷ་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿཝཱ་གཱི་ཤྭ་ར་པ་ཧཱུྃ། ཨོཾ་ཨཱཿཔྲཛྙཱ་ཀཱུ་ཊ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿཨཀྵོ་བྷྱ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཔུཎྱཱ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟི་དྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་པཱ་ཎི་ཧཱུྃ། ཨོཾ་ཨཱཿཔྲ་སེ་ན་ཙནྡྲ་ཧཱུྃ། ཨོཾ་ཨཱཿསུ་བྷཱུ་ཏི་པཱ་ལ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱ་ནནྡ་གརྦྷ་ཧཱུྃ། ཨོཾ་ཨཱཿཔྲཛྙཱ་པཱ་ལི་ཏ་ཧཱུྃ། ཨོཾ་ཨཱཿཤཱནྟི་གརྦྷ་ཧཱུྃ། ཨོཾ་ཨཱཿཤ་ཀུ་ལ་ན་མི་ཏྲ་ཧཱུྃ། ཨོཾ་ཨཱཿསུ་ཁ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཤྲཱདྡྷ་ཀ་ར་ཝརྨ་ཧཱུྃ། ཨོཾ་ཨཱཿརཏྣ་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿགུ་ཎ་ཤཱི་ལ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱནནྡ་གརྦྷ་ཧཱུྃ། ཨོཾ་ཨཱཿཔུཎྱཱ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟི་དྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱནནྡ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་གརྦྷ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་ཌཱ་ཀི་ནཱི་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱ་ལ་ཙ་ཀྲ་པཱ་ད་ཧཱུྃ། ཨོཾ་ཨཱཿཨ་ནུ་པ་མ་རཀྵི་ཏ་ཧཱུྃ། ཨོཾ་ཨཱཿསཱ་དྷ་པུ་ཏྲ་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨཱ་ཀ་ར་ཧཱུྃ།

ཨོཾ་ཨཱཿདྷརྨ་མི་ཏྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབིཀྵ་ཏ་དེ་ཝ་ཧཱུྃ། ཨོཾ་ཨཱཿཤཱཀྱ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿཨཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་པཱ་ཎི་ཧཱུྃ། ཨོཾ་ཨཱཿནཱ་གཱརྫུ་ན་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱརྻ་དེ་ཝ་ཧཱུྃ། ཨོཾ་ཨཱཿཤཱཀྱ་མི་ཏྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཙནྡྲ་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿལི་ལ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབྷ་བྱ་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿཔྲ་བྷ་ཀ་ར་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿཔྲ་བྷ་ཀ་ར་དེ་ཝ་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨཱ་ཀ་ར་ཤཱནྟི་བ་ཧཱུྃ། ཨོཾ་ཨཱཿབིཀྵ་ཏ་དེ་བ་ཧཱུྃ། ཨོཾ་ཨཱཿཤཱཀྱ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨ་སྭ་མྱཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿལ་ལི་ཏ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཨ་མོ་གྷ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཛྙཱ་ན་ཨཱ་ཀ་ར་གུཔྟ་ཧཱུྃ། ཨོཾ་ཨཱཿཔདྨ་བཛྲ་ཧཱུྃ། ཨོཾ་ཨཱཿདཱི་པཾ་ཀ་ར་རཀྵི་ཏ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨཱ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཛྙཱ་ན་སི་དྷ་ཧཱུྃ། ཨོཾ་ཨཱཿལ་ཀྵཾ་སིཾ་ཧ་ཧཱུྃ། ཨོཾ་ཨཱཿཨི་ཤྭ་ར་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿདྭེ་ཥ་ཡ་མཱ་རི་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །།ཨོཾ་ཨཱཿཤཱཀྱ་མུ་ནི་ཧཱུྃ། ཨོཾ་ཨཱཿམཉྫུ་གྷོ་ཥ་ཧཱུྃ། ཨོཾ་ཨཱཿཨ་ཤྭ་ཀ་ཤྲཱི་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲཱ་ས་ན་པཱ་ད་ཧཱུྃ། ཨོཾ་ཨཱཿརཏྣ་ཀཱིརྟི་ཧཱུྃ། ཨོཾ་ཨཱཿཨཱནནྡ་གརྦྷ་ཧཱུྃ། ཨོཾ་ཨཱཿཔུཎྱཱ་གྲ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟི་དྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨ་སྭ་མྱཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ། ཨོཾ་ཨཱཿབཛྲ་དྷཱ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿལོ་ཀེ་ཤྭ་ར་ཧཱུྃ། ཨོཾ་ཨཱཿཤ་ཝ་རི་པ་ཧཱུྃ། ཨོཾ་ཨཱཿམེ་ཏྲཱི་གུ་ཧྱ་ཧཱུྃ། ཨོཾ་ཨཱཿཤཱཀྱ་རཀྵི་ཏ་ཧཱུྃ། ཨོཾ་ཨཱཿཀ་ན་ཀ་ཤྲཱི་ཧཱུྃ། ཨོཾ་ཨཱཿཛ་ཡ་སེ་ན་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨཱ་གུ་ཎ་ཧཱུྃ། ཨོཾ་ཨཱཿཀཱིརྟི་དྷྭ་ཛ་ཧཱུྃ། ཨོཾ་ཨཱཿདྷརྨཱ་སྭ་མྱཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།

ཨོཾ་ཨཱཿམ་ཏི་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲ་ཧཱུྃ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
