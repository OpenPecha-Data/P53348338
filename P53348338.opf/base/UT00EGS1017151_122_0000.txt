༼ཇ༽༄༅། །དཔལ་ཀྱེའི་རྡོ་རྗེའི་དཀྱིལ་འཁོར་གྱི་བཀྲ་ཤིས་བཞུགས།། མཚུངས་མེད་བླ་མ་དཔལ་ལྡན་རྡོ་རྗེ་འཆང༌། །མཐུ་སྟོབས་དབང་ཕྱུག་བླ་མ་བརྒྱུད་པར་བཅས། །མཚན་ནས་བརྗོད་དམ་ཅུང་ཟད་དྲན་ཡང་རུང༌། །བཀྲ་ཤིས་འཕེལ་ཕྱིར་རྟག་ཏུ་དྲན་པར་འགྱུར། །པདྨའི་ལྟེ་བར་བདུད་བཞིའི་སྟེང་ན་བརྗིད། །ཡོན་ཏན་མཐའ་ཡས་ཡེ་ཤེས་ལྔ་ཡི་དངོས། །ཧེ་རུ་ཀ་དཔལ་བདག་མེད་རྣལ་འབྱོར་མ། །བཀྲ་ཤིས་དེས་ནི་སྐྱེ་དགུ་བདེ་ལེགས་ཤོག །སྣ་ཚོགས་པདྨའི་འདབ་མར་བདེ་བར་བཞུགས། །སྐྱེ་མཆེད་འབྱུང་བ་རྣམ་པར་དག་པའི་དངོས། །ཡེ་ཤེས་ཚད་མེད་རང་བཞིན་ལྷ་མོ་བརྒྱད། །བཀྲ་ཤིས་དེས་ནི་སྐྱེ་དགུ་བདེ་ལེགས་ཤོག །ཁམས་དྲུག་དག་པ་ཡེ་ཤེས་ཆེན་པོའི་སྐུ། །ལམ་གྱི་བྱང་ཆུབ་སུམ་ཅུ་རྩ་བདུན་ཏེ། །སྣང་བ་སངས་རྒྱས་ཞིང་འགྱུར་གཞལ་ཡས་ཁང༌། །བཀྲ་ཤིས་དེས་ནི་སྐྱེ་དགུ་བདེ་ལེགས་ཤོག །ཡན་ལག་རྣམ་དག་ཁྲོ་བོ་ཆེན་པོའི་སྐུ། །རྡོ་རྗེའི་ས་གཞི་རྭ་བ་བླ་རེ་གུར། །བདུད་དཔུང་ངེས་འཇོམས་སྲུང་བའི་འཁོར་ལོ་མཆོག །བཀྲ་ཤིས་དེས་ནི་སྐྱེ་དགུ་བདེ་ལེགས་ཤོག །དག་བྱ་དག་བྱེད་ཕྱིའམ་ནང་ན་བཞུགས། །ཕྱོགས་སྐྱོང་ཞིང་སྐྱོང་ཀླུ་དང་སྤྲིན་དང་ཤིང༌། །མཆོད་རྟེན་རོ་དང་འདྲེ་དང་མེ་དང་ཆུ། །ཕྱོགས་བརྒྱད་དག་བྱེད་དེས་ནི་བཀྲ་ཤིས་ཤོག །བླ་མ་དཀྱིལ་འཁོར་གཙོ་བོ་འཁོར་དང་བཅས། །སྦྱང་གཞི་དག་བྱེད་ལམ་དང་འབྲས་བུ་མཆོག །བརྗོད་དམ་དེ་དོན་རྟག་ཏུ་དྲན་ཡང་རུང༌། །བཀྲ་ཤིས་འཕེལ་ཕྱིར་རྟག་ཏུ་དྲན་པར་འགྱུར། །བཀྲ་ཤིས་ཀྱི་ཚིགས་སུ་བཅད་པ་དགེ་བསྙེན་གྲགས་པ་རྒྱལ་མཚན་གྱིས་སྦྱར་བ་རྫོགས་སོ།། །།ཨོཾ་ཤྲཱི་བཛྲ་ཧེ་རུ་ཀ་ས་མ་ཡ་མ་ནུ་པཱ་ལ་ཡ། ཧེ་རུ་ཀ་ཏྭེ་ནོ་པ་ཏིཥྛ། དྲྀཌྷོ་མེ་བྷ་ཝ། སུ་ཏོ་ཥྱོ་མེ་བྷ་ཝ། ཨ་ནུ་རཀྟོ་མེ་བྷ་ཝ། སུ་པོ་ཥྱོ་མེ་བྷ་ཝ། སརྦ་སིདྡྷིམྨེ་པྲ་ཡཙྪ། སརྦ་ཀརྨ་སུ་ཙ་མེ། ཙིཏྟཾ་ཤྲེ་ཡཿཀུ་རུ་ཧཱུྃ། ཧ་ཧ་ཧ་ཧ་ཧོཿབྷ་ག་ཝཱན་བཛྲ་ཧེ་རུ་ཀ་མཱ་མེ་མུཉྩ་ཧེ་རུ་ཀོ་བྷ་ཝ།

མ་ཧཱ་ས་མ་ཡ་སཏྭ་ཨ་ཧཱུྃ་ཕཊ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
