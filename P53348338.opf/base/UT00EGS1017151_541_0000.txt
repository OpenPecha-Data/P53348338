༼བ༽༄༅། །འཇམ་དཔལ་ལ་མཚན་དོན་གྱི་སྒོ་ནས་བསྟོད་པ་བཞུགས།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། ན་མོ་མཉྫུ་གྷོ་ཥཱ་ཡ། སྐྱབས་མཆོག་སངས་རྒྱས་ཀུན་གྱི་དངོས། །དཔལ་ལྡན་བླ་མ་ལ་བཏུད་དེ། །སངས་རྒྱས་ཐམས་ཅད་བསྐྱེད་པའི་ཡབ། །འཇམ་པའི་དཔལ་ལ་བསྟོད་པར་བགྱི། །མིང་ལ་བརྟེན་ནས་དོན་ཚོལ་བ། །འདི་ནི་ཤེས་ལྡན་རྣམས་ཀྱི་ལུགས། །ཐུབ་པས་ཁྱོད་མཚན་འཇམ་དཔལ་ཞེས། །བསྒྲགས་དོན་དེ་ནི་འདི་ལྟར་ཁུམས། །སེམས་ཀྱི་རང་བཞིན་སྤྲོས་པ་ཡི། །གྲྭ་ཟུར་ཞི་བས་འཇམ་པ་ཉིད། །རྟེན་འབྱུང་སྲིད་ཞིའི་དཔལ་གྱི་གཞི། །འདི་འདྲ་རྒྱུ་ཡི་འཇམ་དཔལ་ལགས། །སྟོང་ཉིད་རྟོགས་པས་ལྟ་བ་ཡི། །ཚང་ཚིང་ཞི་བས་འཇམ་པ་ཉིད། །སྙིང་རྗེས་སྐྱེ་འགྲོའི་དཔལ་གྱུར་པ། །འདི་འདྲ་ལམ་གྱི་འཇམ་དཔལ་ལགས། །སྐུ་གསུམ་མི་ཕྱེད་མཉམ་སྦྱོར་ཞིང༌། །སྐུ་གསུང་ཐུགས་ཀྱི་རྡོ་རྗེ་ཅན། །ཡོན་ཏན་ཕྲིན་ལས་མཐར་ཕྱིན་པ། །འདི་འདྲ་འབྲས་བུའི་འཇམ་དཔལ་ལགས། །ཆོས་དབྱིངས་རང་བཞིན་འཇམ་པའི་ངང༌། །གཟུགས་ཀྱི་སྐུ་ནི་གདུལ་བྱའི་དཔལ། །སྐུ་གསུམ་ལྷུན་གྱིས་གྲུབ་པ་པོ། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །ཀུན་ནས་མཛེས་པ་འཇམ་པའི་སྐུ། །མཚན་དང་དཔེ་བྱད་ཉེར་འབར་བ། །གང་གིས་མཐོང་བའི་དཔལ་གྱུར་པ། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །ཐེག་མཆོག་སྟོན་པ་འཇམ་པའི་དབྱངས། །ཡན་ལག་དྲུག་ཅུས་ཟུག་རྔུ་སེལ། །གང་གིས་ཐོས་པས་དཔལ་གྱུར་པ། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །སྒྲིབ་གཉིས་དྲི་བྲལ་འཇམ་པའི་ཐུགས། །ཡེ་ཤེས་ལྔ་ཡི་བདག་ཉིད་ཅན། །གང་གིས་དྲན་པའི་དཔལ་གྱུར་པ། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །སྲིད་ཞི་མཉམ་ཉིད་འཇམ་པ་སྟེ། །སྟོབས་དང་མི་འཇིགས་རིག་པ་སོགས། །གང་གིས་རྗེས་སུ་སློབ་པའི་དཔལ། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །འགྲོ་ལ་མཉམ་དགོངས་འཇམ་པ་ཉིད། །རྣམ་འཕྲུལ་དུ་མས་རྣམ་ཀུན་ཏུ། །གང་གིས་དད་པའི་དཔལ་གྱུར་པ། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ལྟར་རྒྱུ་དང་ལམ་དང་ནི། །འབྲས་བུའི་ཆོས་རྣམ་གང་ལགས་པ། །དེ་ཀུན་འཇམ་པའི་དཔལ་ཉིད་དེ། །རྣམ་པར་སྤྲོས་ན་མཐའ་ཡས་ལགས། །དེ་སླད་དུས་གསུམ་རྒྱལ་བ་ཀུན། །འཇམ་དཔལ་གྱིས་བསྐྱེད་ཅེས་པ་དང༌། །སངས་རྒྱས་ཀུན་གྱི་ཡེ་ཤེས་སྐུ། །གཅིག་ཏུ་བསྡུས་པ་འཇམ་དཔལ་ཞེས། །གསུངས་པའང་ངེས་པའི་དོན་དུ་འཚལ། །རྒྱལ་བ་མིང་དང་མཚན་མ་ཡི། །ཚུལ་དུ་གང་བཏགས་དེ་ཡིས་ནི། །དེ་བཞིན་གཤེགས་པ་མ་མཐོང་ལ། །གང་གིས་ཚུལ་འདིས་སངས་རྒྱས་ཀུན། །འཇམ་དཔལ་ཉིད་དུ་ལེགས་རྟོགས་པ། །དེ་ཡིས་རྒྱལ་བའི་ངེས་པའི་སྐུ། །གདོན་མི་ཟ་བར་འཚལ་བ་ལགས། །དེ་སླད་གངྒཱའི་རྡུལ་སྙེད་ཀྱི། །ཐུང་མཚན་བསྐལ་པར་བརྗོད་པ་བས། །འཇམ་དཔལ་ཞེས་བརྗོད་ཁྱད་འཕགས་ཞེས། །གང་གསུངས་དེ་ཡང་ངེས་པར་ཁུམས། །རྒ་དང་འཆི་བའི་འཇིགས་བྲལ་དང༌། །ཚངས་སྤྱོད་ཆོས་ཀུན་གནས་མེད་དང༌། །གཞན་གྱི་དོན་ལ་རྩེ་དགའི་ཕྱིར། །གཞོན་ནུར་གྱུར་པ་ཁྱོད་ཕྱག་འཚལ། །བདུད་དང་གཤིན་རྗེའི་འཇིགས་པ་ཀུན། །མ་རིག་པ་ལས་བྱུང་བ་སྟེ། །ཁྱོད་ཀྱིས་དེ་ཡང་དབྱིངས་འབྱིན་པས། །གཤིན་རྗེའི་དགྲར་གྱུར་ཁྱོད་ཕྱག་འཚལ། །སྐུ་ཡི་མཚན་དང་རྣམ་པ་དང༌། །ཁ་དོག་དང་ནི་ཕྱག་མཚན་དང༌། །ཆ་ལུགས་གསུང་གིས་གསང་སྔགས་ཀུན། །གདུལ་བྱའི་རྒྱུད་ལ་ཟབ་མོའི་དོན། །མཚོན་པའི་ཕྱིར་དང་ཐོབ་བྱའི་ཕྱིར། །སྒྱུ་མའི་ཚུལ་གྱིས་རྣམ་སྤྲུལ་པ། །འདི་འདྲ་རྒྱལ་བ་ཐབས་མཁས་པ། །བསམ་མི་ཁྱབ་པའི་རྣམ་འཁྲུལ་ལགས། །དེ་སླད་དེ་ལ་རྣམ་ཀུན་ཏུ། །སྤྱི་བོས་གུས་པས་ཕྱག་འཚལ་ཞིང༌། །བསམ་པ་ཐག་པས་སྐྱབས་སུ་མཆི། །རྟག་ཏུ་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། །ཁྱོད་ཀྱི་རང་བཞིན་ཇི་བཞིན་དང༌། །རྣམ་འཕྲུལ་མཛད་པ་ཇི་སྙེད་པ། །འཕགས་མཆོག་རྣམས་ཀྱིའང་ཡུལ་མིན་ན། །བདག་འདྲས་ཇི་ལྟར་རྟོགས་པར་འགྱུར། །འོན་ཀྱང་བསམ་མི་ཁྱབ་པ་དང༌། །མཚུངས་མེད་དཔེ་མེད་ཟླ་མེད་ཅིང༌། །དཔག་ཡས་ཉིད་དུ་ལེགས་ཁུམས་པས། །དེ་ལྟའི་ཚུལ་གྱིས་བསྔགས་པ་ལགས། །ཁྱོད་ཀྱི་མཚན་ལས་ཁྱོད་ཡོན་ཏན། །བརྟགས་ནས་ཚིག་གི་ཕྲེང་བས་བསྟོད། །དོན་དང་མི་ལྡན་རྗེས་བསྔགས་པ། །སྒྲོ་འདོགས་ཡིན་ཕྱིར་སྨད་པའི་གནས། །འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་མཚན་དོན་གྱི་སྒོ་ནས་བསྟོད་པ། ཚིགས་སུ་བཅད་པ་ཉི་ཤུ་རྩ་ལྔ་པ་འདི་ནི། ཆོས་ཀྱི་རྒྱལ་པོ་དཔལ་ལྡན་ས་སྐྱ་པཎྜི་ཏ་ཆེན་པོའི་ཞབས་རྡུལ་སྤྱི་བོས་ལེན་པ། འཕགས་པའི་མིང་དང་ལྡན་པས། མེ་མོ་སྦྲུལ་གྱི་ལོ་དབྱར་ཟླ་འབྲིང་པོའི་ཚེས་བཅུ་བདུན་ལ་རི་བོ་རྩེ་ལྔར་སྦྱར་བའོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
