།འུར་འུར་ཆིལ་ཆིལ་སྒྲ་སྒྲོགས་ཤིང༌། །མེ་སྟག་སྐར་མདའ་ལྟ་བུར་འཕྲོ། །ཁ་དོག་སྣ་ཚོགས་ཀུན་ཏུ་འཁྲུག །བདུད་དཔུང་ངེས་པར་སྲེག་བྱེད་པ། །ས་དང་ནམ་མཁའ་ཀུན་ཁྱབ་པའི། །འབར་བའི་མེ་དཔུང་ཆེན་པོར་གྱུར། །ཨོཾ་བཛྲ་ཛྭཱ་ལ་ཨ་ན་ལཱརྐ་ཧཱུྃ་རྃ་ཧཱུྃ། ཧེ་རུ་ཀ་དཔལ་རྒྱལ་པོ་ཡི། །གཉིས་མེད་ཡེ་ཤེས་ལས་བྱུང་བའི། །སྲུང་བའི་དམ་ཚིག་སྦྱོར་བ་ཡིས། །རྣལ་འབྱོར་རྟག་ཏུ་སྲུང་བར་བྱེད། །ཨོཾ་བཛྲ་རཀྵ་རཀྵ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ། དེ་ནས་གདུག་པ་བསྒྲལ་བའི་ཕྱིར། །ཕྱོགས་བཅུའི་རྒྱལ་བ་ཐམས་ཅད་ཀྱིས། །དཔའ་བོ་ཆེན་པོ་རབ་བསྐུལ་བས། །ཐུགས་ཀའི་ཧཱུྃ་གི་ཡི་གེ་ལས། །འབར་བའི་ཧཱུྃ་བཅུ་རབ་འཕྲོས་ནས། །ཧཱུྃ་ཡིག་དེ་དག་གྱུར་པ་ལས། །ཁམས་གསུམ་འཇོམས་པའི་ནུས་པ་ཅན། །བསྐལ་པའི་མེ་ལྟར་འོད་འཕྲོ་བའི། །རྡོ་རྗེ་ཕུར་བུ་རྣམ་བཅུར་གྱུར། །ཨོཾ་བཛྲ་ཀཱི་ལི་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། སླར་ཡང་ཧཱུྃ་ལས་ཧཱུྃ་བཅུ་འཕྲོས། །ཧཱུྃ་བཅུ་ཡོངས་སུ་གྱུར་པ་ལས། །སྲིད་གསུམ་འཇིག་བྱེད་ནུས་པ་ཅན། །རྡོ་རྗེ་ཐོ་བ་རྣམ་བཅུར་གྱུར། །ཨོཾ་བཛྲ་མུཏྐ་ར་ཧཱུྃ་ཕཊ། ཧཱུྃ་ཕྱོགས་བཅུར་གནས་པའི་བགེགས་ཀྱི་ཚོགས། །ཐམས་ཅད་ང་ལ་མཉན་པར་གྱིས། །ལྷ་དང་ལྷ་མིན་གྲུལ་བུམ་སོགས། །ཞི་ཞིང་ཞི་བའི་སེམས་ལྡན་པ། །དེ་དག་རྡོ་རྗེ་འཛིན་པའི་བཀས། །དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་སོང༌། །བྱང་ཆུབ་སེམས་ལ་གནས་པར་གྱིས། །གནོད་སྦྱིན་སྲུལ་པོ་གང་ཅིའང་རུང༌། །གདུག་ཅིང་དུག་པའི་སེམས་ལྡན་པ། །དེ་དག་མངོན་སྤྱོད་ལས་ཀྱིས་ནི། །ད་ལྟ་ཉིད་དུ་བསྒྲལ་བར་བྱ། །ཧཱུྃ་ཕྱོགས་བཅུར་སྲུང་བའི་ཕོ་ཉའི་ཚོགས། །བགེགས་ཀྱི་ཚོགས་རྣམས་གཞོམ་པའི་ཕྱིར། །རྡོ་རྗེ་ཕུར་བུ་འདི་དག་དང༌། །ཐོ་བ་འདི་དག་བླང་བར་གྱིས། །དེ་ནས་རྡོ་རྗེ་ཕོ་ཉ་བཅུས། །ཕྱག་གཡོན་དང་པོས་ཕུར་བུ་བླངས། །གཡས་ཀྱི་དང་པོས་ཐོ་བ་བསྣམས། །འོག་གཡས་ཅང་ཏེའུ་ཁ་ཊྭཱཾ་འཛིན། །སྤྲུལ་པས་ནམ་མཁའི་ཁམས་བཀང་ནས། །དྲག་པོའི་ལས་ལ་བརྩོན་པར་གྱུར། །ཧཱུྃ་ཤར་ཕྱོགས་སྲུང་བའི་ཁྭ་གདོང་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །ཤར་གྱི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཀཱ་ཀཱ་སྱེ།

ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ། བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ་བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་བྱང་ཕྱོགས་སྲུང་བའི་འུག་གདོང་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །བྱང་གི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཨུ་ལཱུ་ཀཱ་སྱེ། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ། བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་ནུབ་ཕྱོགས་སྲུང་བའི་ཁྱི་གདོང་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །ནུབ་ཀྱི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཤྭ་ནཱ་སྱེ། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ། བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ།

ཨོཾ་བཛྲ་མུཏྐ་ར་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་ལྷོ་ཕྱོགས་སྲུང་བའི་ཕག་གདོང་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །ལྷོ་ཡི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། ཨོཾ་སུ་ཀ་རཱ་སྱེ། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ། བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ། ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་ཤར་ལྷོ་སྲུང་བའི་བརྟན་མ་ནི། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །ཤར་ལྷོའི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཡ་མ་དཱ་དྷི། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷཱ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ། བཛྲ་ཀཱི་ལ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་ལྷོ་ནུབ་སྲུང་བའི་ཕོ་ཉ་མོ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །ལྷོ་ནུབ་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཡ་མ་དཱུ་ཏི་ཡེ།

ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ། བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀ་ཡ་ཝཱཀ་ཙིཏྟ་བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ། ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་ནུབ་བྱང་སྲུང་བའི་མཆེ་བ་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །ནུབ་བྱང་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཡ་མ་དཾཥྚཱི་ནི། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ། བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ལ་ཝཱཀ་ཙིཏྟ་བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་བྱང་ཤར་སྲུང་བའི་འཇོམས་མ་ནི། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །བྱང་ཤར་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཡ་མ་མ་ཐ་ནི། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ། བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན། ཀཱ་ལ་ཝཱཀ་ཙིཏྟ་བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ།

ཧཱུྃ་སྟེང་ཕྱོགས་སྲུང་བའི་དུམ་སྐྱེས་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །སྟེང་གི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ཏེ། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཁཎྜ་རོ་ཧཱ། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ་བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། ཧཱུྃ་འོག་ཕྱོགས་སྲུང་བའི་དུམ་སྐྱེས་མ། །སྤྲུལ་པ་དུ་མ་དང་བཅས་པས། །འོག་གི་ཕྱོགས་ཀྱི་བགེགས་ཚོགས་ཀུན། །རིག་སྔགས་ཞགས་པས་མྱུར་དུ་ཆིངས། །རྡོ་རྗེའི་ལྟ་བས་རེངས་པར་གྱིས། །ཧཱུྃ་གི་སྒྲ་ཡིས་བརྒྱལ་བྱས་ནས། །འབར་བའི་རྡོ་རྗེ་ཕུར་བུ་ཡིས། །སྤྱི་བོའི་གཙུག་ནས་དགས་པར་གྱིས། །ཨོཾ་ཁཎྜ་རོ་ཧཱ། ཨོཾ་གྷ་གྷ་གྷཱ་ཏ་ཡ་གྷཱ་ཏ་ཡ། སརྦ་དུཥྚཱ་ན་ཕཊ། ཀཱི་ལ་ཡ་ཀཱི་ལ་ཡ། སརྦ་པཱ་པཾ་ཧཱུྃ་ཕཊ། བཛྲ་ཀཱི་ལ་ཡ་བཛྲ་དྷ་ར་ཨཱ་ཛྙཱ་པ་ཡ་ཏི། སརྦ་བིགྷྣཱན་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ། བཛྲ་ཀཱི་ལ་ཡ་ཧཱུྃ་ཕཊ། གཡས་པའི་རྡོ་རྗེ་ཐོ་བ་ཡིས། །རྡོ་རྗེ་ཕུར་བུ་རབ་བསྣུན་ནས། །རྡུལ་ཕྲན་ཙམ་དུ་རླག་པར་གྱིས། །རྣམ་ཤེས་ཆོས་ཀྱི་དབྱིངས་སུ་སྒྲོལ། །ཨོཾ་བཛྲ་མུཏྐ་ར་བཛྲ་ཀཱི་ལ་ཡ་ཨཱ་ཀོ་ཊ་ཡ་ཧཱུྃ་ཕཊ། རྡོ་རྗེ་ཕོ་ཉ་ཐམས་ཅད་ཀྱིས། །སྒྲུབ་པོས་བྱང་ཆུབ་ཐོབ་ཀྱི་བར། །མི་མཐུན་ཕྱོགས་རྣམས་ཐམས་ཅད་ལས། །སྒྲུབ་པོ་རྟག་ཏུ་སྲུང་བར་གྱིས། །རྡོ་རྗེ་འཛིན་པ་འཕགས་པས་སྦྱར་བའོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
