༼བ༽༄༅། །ཛམྦྷ་ལ་སེར་བོའི་སྒྲུབ་ཐབས་བཞུགས།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། བླ་མ་དང་དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །འཕགས་པ་ཛམྦྷ་ལ་བསྒོམ་པར་འདོད་པས། སྟན་བདེ་བ་ལ་བསྡད། བླ་མ་དང་དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་འགྲོ་བ་བྱ། སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དུ་སངས་རྒྱས་ཐོབ་པར་བྱ། གནས་སྐབས་སུ་བདག་དང་གཞན་གྱི་དབུལ་བ་དང་རྒུད་པ་ཐམས་ཅད་བསལ་ནས་འབྱོར་པ་ཕུན་སུམ་ཚོགས་པ་ཐོབ་པར་བྱ། དེའི་དོན་དུ་ཛམྦྷ་ལ་བསྒོམ་པར་བྱ་སྙམ་དུ་བྱང་ཆུབ་ཀྱི་སེམས་བསྒོམ། དེ་ནས་རང་གི་སྙིང་གི་ནང་དུ་ཟླ་བའི་སྟེང་དུ་ཛཾ་གསེར་གྱི་མདོག་ཅན་ཞིག་བསམས་པ་ལས་འོད་འཕྲོས་པས་རང་གི་སྡིག་པ་དང༌། སྒྲིབ་པ་དང༌། དབུལ་བ་དང༌། རྒུད་པ་བསལ། ཡང་བ་སྤུའི་བུ་ག་ནས་འོད་འཕྲོས་པས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་སྡིག་པ་དང༌། སྒྲིབ་པ་སྦྱངས། དབུལ་བ་དང་རྒུད་པ་བསལ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་དཔལ་འབྱོར་བསྡུས་པ་རང་ལ་ཐིམ་པས། རང་ཉིད་རིན་པོ་ཆེའི་གཏེར་སྣ་ཚོགས་ཀྱིས་གང་བའི་བུམ་པའི་ཚོགས་ཀྱིས་བཏེག་པའི་རིན་པོ་ཆེའི་ཁྲིའི་སྟེང་དུ་པདྨ་དང༌། ཟླ་བ་ལ་བཞུགས་པའི་ཛམྦྷ་ལ་གསེར་བཙོ་མ་ལྟ་བུའི་ཞབས་གཡས་ཅུང་ཟད་བརྐྱང་ཞིང༌། གཡོན་བསྐུམ་པའི་ཚུལ་གྱིས་བཞུགས་པ། ཞལ་གཅིག་ཕྱག་གཉིས་པ། སྤྱན་གསུམ་པ། རབ་ཏུ་བརྗིད་ཅིང་སྐུ་ཤ་རྒྱས་པ། ཕྱག་གཡས་པ་མཆོག་སྦྱིན་པས་བི་ཛ་པཱུ་རའི་ཤིང་ཏོག་འཛིན་ཞིང༌། གཡོན་པས་རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱི་རྒྱུན་སྐྱུགས་པའི་ནེའུ་ལེ་འཛིན་པ། རིན་པོ་ཆེའི་རྒྱན་སྣ་ཚོགས་པས་བརྒྱན་ཞིང༌། དར་སྣ་ཚོགས་ཀྱི་ན་བཟའ་མནབས་ལ། སྤྱི་བོར་སངས་རྒྱས་རིན་ཆེན་འབྱུང་ལྡན་གྱིས་དབུར་བརྒྱན་པར་བསམ། དེའི་ཐུགས་ཀའི་ཛྃ་ལས་འོད་འཕྲོས་པས། རི་རབ་ཀྱི་སྟེང་ནས་ཛམྦྷ་ལ་སྤྱན་དྲང༌། ཛཿཧཱུྃ་བྃ་ཧོཿས་རང་ལ་གཉིས་སུ་མེད་པར་བསྟིམ། རང་གི་འཁོར་དུ་འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་ནོར་ལྷ་སྤྲིན་འཐིབས་པ་བཞིན་དུ་བཞུགས་པས་རིན་པོ་ཆེ་དང༌།

ཟས་ནོར་གྱི་ཆར་འབེབས་པར་བསམས་ལ། སེམས་མཉམ་པར་གཞག་གོ །སྔགས་ཟློ་བར་འདོད་ན། རང་གི་སྙིང་གའི་ཛྃ་གྱི་མཐའ་མར་སྔགས་ཀྱི་ཕྲེང་བས་འཁོར་བ་ལས་འོད་འཕྲོས་པས་འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་ནོར་གྱི་ལྷ་ཐམས་ཅད་དབང་དུ་བསྡུས་ནས། དེ་དག་གིས་བདག་ལ་རིན་པོ་ཆེ་དང་ཟས་ནོར་གྱི་ཆར་ཕབ་པར་བསམ་ཞིང༌། ཨོཾ་ཛམྦྷ་ལ་ཛ་ལེནྡྲཱ་ཡ་སྭཱ་ཧཱ། ཞེས་ཅི་ནུས་པ་བཟླའོ། །ཆུ་སྦྱིན་གཏོང་བར་འདོད་ན། རིན་པོ་ཆེའི་སྡེར་གྱི་ནང་དུ་རིན་པོ་ཆེ་ལས་བཞེངས་པའི་ཛམྦྷ་ལ་བཞུགས་སུ་བཅུག་པའམ། བཞུགས་པར་མོས་ལ་བདག་བསྐྱེད་བཞིན་བསྐྱེད་ནས། རིན་པོ་ཆེའི་བུམ་པ་ཆུས་བཀང་ལ། ཨ་མྲྀ་ཏས་བསང༌། སྭ་བྷཱ་ཝས་སྦྱང༌། སྟོང་པའི་ངང་ལས་བྷྲཱུྃ་ལས་རིན་པོ་ཆེའི་བུམ་པའི་ནང་དུ། ཨོཾ་ཨཱཿཧཱུྃ་ལས་བདུད་རྩིའི་ཆུ་རྒྱུན་གྱིས་གང་བ་བསྐྱེད་ལ། རང་གི་ལག་པ་གཡས་པའི་མཐིལ་དུ་ཡི་གེ་ཛྃ་བསྐྱེད་པ་ལས། བདུད་རྩིའི་ཆུ་རྒྱུན་བྱུང་བས། ཛམྦྷ་ལ་ལ་མཆོད་པར་བསམ་ཞིང༌། ལག་པ་གཡས་པའི་ཁྱོར་མོས་ཆུ་ཕུལ། ཨོཾ་ཛམྦྷ་ལ་ཛ་ལེནྡྲཱ་ཡ་སྭཱ་ཧཱ། ཞེས་ཅི་རིགས་པ་བཟླས་པའི་མཐར། ཨོཾ་ཛལུྃ་སྭཱ་ཧཱ། ཞེས་བྱ་བའི་སྔགས་ཀྱིས་ཕུལ་བས་བཅོམ་ལྡན་འདས་མཉེས་ནས་བདག་ལ་འདོད་དགུའི་དངོས་གྲུབ་གནང་བར་བསམས་ལ། ཨོཾ་ཛམྦྷ་ལ་ཛ་ལེནྡྲཱ་ཡ་ཨརྒྷཾ། པཱ་དྱཾ། པུཥྤེ། དྷཱུ་པེ། དཱི་པེ། གནྡྷེ། ནཻ་ཝི་དྱེ་ཨཱཿཧཱུྃ་ཞེས་པས་མཆོད་ལ། བཅོམ་ལྡན་གང་ཞིག་རྒྱལ་བ་ཐམས་ཅད་ཀྱི། །ནམ་མཁའ་མཛོད་ཀྱི་ཏིང་འཛིན་ལས་འཕྲུལ་པས། །གདུལ་བྱ་རྣམས་ལ་དཔལ་འབྱོར་ཀུན་སྩོལ་བ། །རྨུགས་འཛིན་ཆུ་དབང་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །ཞེས་པས་བསྟོད་ནས། སྐུ་གཟུགས་མེད་ན། བཛྲ་མུཿས་གཤེགས་སུ་གསོལ། ཡོད་ན་བསྡུས་ལ་ཆུ་ས་གཙང་མའམ། ཆུ་ར་ལ་སོགས་པར་དོར་རོ། །ཆུ་སྦྱིན་གྱི་ཆོ་གའོ། །འཕགས་པ་ཛམྦྷ་ལའི་གྲུབ་ཐབས་མདོར་བསྡུས་པ་འདི་ནི་ཀིམ་ཀཾ་ནུ་ཞེས་བྱ་བས་བསྐུལ་བའི་དོན་དུ་འཕགས་པས་ས་མོ་ཡོས་བུའི་ལོ་སྟོན་ཟླ་འབྲིང་པོའི་ཚེས་བཅུ་དགུ་ལ་དཔལ་ས་སྐྱར་སྦྱར་བའོ།།

།། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
