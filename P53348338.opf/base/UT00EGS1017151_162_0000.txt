༼ཉ༽༄༅། །སྒྲོལ་མ་ཉིན་ཞི་མཚན་ཁྲོའི་སྒྲུབ་ཐབས་བཞུགས།། ༄༅། །རྗེ་བཙུན་མ་འཕགས་མ་སྒྲོལ་མ་ལ་ཕྱག་འཚལ་ལོ། །རྗེ་བཙུན་མ་སྒྲོལ་མ་ཞི་ཁྲོ་གཉིས་བསྒོམ་པར་འདོད་པའི་རྣལ་འབྱོར་པས། ཐོ་རངས་ལངས་ལ་ཁ་དག་པ་ལ་སོགས་པའི་གཙང་སྦྲ་བྱས་ཏེ། དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་འགྲོ་བ་དང་བྱང་ཆུབ་ཀྱི་སེམས་བསྒོམས་ལ། མདུན་དུ་པདྨའི་སྡོང་པོ་ཆེན་པོ་གཅིག་གི་སྟེང་དུ་ཨ་ལས་ཟླ་བའི་དཀྱིལ་འཁོར་དེའི་སྟེང་དུ་ཧྲཱིཿདཀར་པོ། དེ་གྱུར་པ་ལས་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་སྐུ་མདོག་དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ། གཡས་མཆོག་སྦྱིན། གཡོན་པདྨ་འཛིན་པའི་པདྨ་ཐུགས་ཀར་ཁ་ཕྱེ་བའི་སྟེང་ན། རྗེ་བཙུན་མ་སྒྲོལ་མ་ལྗང་གུ་ཞི་བའི་སྐུ་བདག་ལ་ཞལ་གཟིགས་པར་བསམ། དེ་ལ་མེ་ཏོག་ལ་སོགས་པ་མཆོད་པ་ལྔས་མཆོད། དཀོན་མཆོག་གསུམ་ལ་བདག་སྐྱབས་མཆི། །སྡིག་པ་ཐམས་ཅད་སོ་སོར་བཤགས། །འགྲོ་བའི་དགེ་ལ་རྗེས་ཡི་རང༌། །སངས་རྒྱས་བྱང་ཆུབ་ཡིད་ཀྱིས་གཟུང༌། །སངས་རྒྱས་ཆོས་དང་ཚོགས་མཆོག་ལ། །བྱང་ཆུབ་བར་དུ་སྐྱབས་སུ་མཆི། །རང་གཞན་དོན་ནི་རབ་བསྒྲུབ་ཕྱིར། །བྱང་ཆུབ་སེམས་ནི་བསྐྱེད་པར་བགྱི། །བྱང་ཆུབ་མཆོག་གི་སེམས་ནི་བསྐྱེད་བགྱིས་ནས། །སེམས་ཅན་མ་ལུས་བདག་གིས་མགྲོན་དུ་གཉེར། །བྱང་ཆུབ་སྤྱོད་མཆོག་ཡིད་འོང་སྤྱད་པར་བགྱི། །འགྲོ་ལ་ཕན་ཕྱིར་སངས་རྒྱས་འགྲུབ་པར་ཤོག །ཅེས་པ་ལ་སོགས་པས་ཡན་ལག་བདུན་པ་བྱ། དེ་ནས་དེ་ལ་བསྟོད་པ་ཉི་ཤུ་རྩ་གཅིག་བརྗོད་པས་བསྟོད། ཚར་གཉིས་པ་བརྗོད་པས་བདག་ལ་བསྟིམ་ལ། བདག་ཀྱང་རྗེ་བཙུན་མ་སྒྲོལ་མ་ལྗང་གུ་ཞི་བའི་སྐུ། ཞལ་གཅིག་ཕྱག་གཉིས་པ་གཡས་མཆོག་སྦྱིན། གཡོན་པའི་མཐེབ་སྲིན་གྱིས་ཨུཏྤ་ལའི་རྩ་བ་བཟུང་ནས་མེ་ཏོག་འདབ་མ་རྣ་བའི་དྲུང་དུ་ཁ་ཕྱེ་བ། པདྨ་དང་ཟླ་བའི་གདན་ལ་སྐྱིལ་ཀྲུང་ཕྱེད་པ་གཡས་བརྐྱང་པས་བཞུགས་པ་ཟླ་བའི་རྒྱབ་ཡོལ་ཅན། རིན་པོ་ཆེའི་རྒྱན་གྱིས་བརྒྱན་པ། དར་གྱི་ཤམ་ཐབས་ཅན།

དབུ་སྐྲ་ལི་བ། གཡོན་གྱི་ཕྱེད་དུ་བཅིངས་པ། གཡས་ཅུང་ཟད་ཅིག་འཕྱང་བ། སྤྱི་བོ་ན་ཟླ་བ་ཚེས་པའི་སྟེང་ན་འོད་དཔག་ཏུ་མེད་པས་དབུར་བརྒྱན་པར་བསམ་མོ། །བསྟོད་པ་ཚར་གསུམ་པ་འདོན་ཅིང་བདག་རང་གསལ་གདབ། དེ་ནི་གཉིས་གསུམ་ཞེས་པའི་དོན་ཡིན་ནོ། །དེ་ནས་བསྟོད་པ་འདོན་པར་འདོད་ན། བདུན་དུ་མངོན་པར་བརྗོད་ན། ཞེས་བྱ་བའི་དོན་ནི། ལྷའི་སྐུ་ལ་དམིགས་ལ་བདུན་ལ་སོགས་པ་ཇི་ལྟར་རིགས་པ་གདོན་པར་བྱའོ། །གསང་སྔགས་ཟློས་ན་སྙིང་གར་པདྨ་འདབ་མ་བརྒྱད་ལ་དབུས་སུ་ཨོཾ་དང་ཧཱའི་བར་དུ་རང་མང་ལ་རཀྵཿལ་སོགས་པའི་སྤེལ་ཚིག་ཇི་ལྟར་འདོད་པ་སྤེལ། འདབ་མ་རྣམས་ལ་སྔགས་ལྷག་མ་བཀོད་ལ། དེ་ལ་དམིགས་ལ་ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་ཧཱ། ཞེས་བཟླ་བར་བྱའོ། །སྲོད་ཀྱི་དུས་སུ་ཆོ་ག་ལྷག་མ་ཐམས་ཅད་སྔ་མ་དང་འདྲ་མོད་ཀྱི། མི་འདྲ་བའི་ཁྱད་པར་ནི་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་གི་པདྨའི་སྟེང་ན་བཞུགས་པའི་རྗེ་བཙུན་མ་ཁྲོ་མོའི་སྐུ་རུ་བསྒོམ། དེ་བདག་ལ་བསྟིམ་ལ། བདག་གསལ་འདེབས་པའི་དུས་སུ་བདག་ཀྱང་ཁྲོ་མོའི་སྐུ་ཞལ་གཅིག་ཕྱག་གཉིས་མ་སྐུ་མདོག་དཀར་མོ་ཕྱག་གཡས་ན་རྡོ་རྗེ་རྩེ་ལྔ་པ་འཕྱར་བའི་ཚུལ་དུ་འཛིན་པ། ཕྱག་གཡོན་པ་ས་ལ་བརྡབས་པས་འཇིག་རྟེན་གསུམ་པོ་གཡོ་བར་བྱས་ཏེ་སྡིགས་མཛུབ་ཏུ་བྱས་ནས་ཐུགས་ཀར་འཛིན་པ། དབུ་སྐྲ་དམར་སེར་གྱེན་དུ་བརྫེས་པ་རབ་ཏུ་ཁྲོས་པའི་སྤྱན་དམར་ཞིང་ཟླུམ་པ། ཞལ་མཆེ་བ་གཙིགས་པ། སྟག་གི་པགས་པའི་ཨཾ་ག་རག་ཏུ་བྱས་པའི་སྟོད་གཡོགས་གཟིག་གི་པགས་པའི་ཤམ་ཐབས་ཀྱིས་ལོང་བུའི་བར་དུ་སླེབ་པ། རིན་པོ་ཆེའི་རྒྱན་གྱིས་བརྒྱན་པ། སྦྲུལ་ལྗང་གུས་སེ་རལ་ཁ་ཐོགས་པ། པདྨའི་གདན་ལ་བཞེངས་ནས་བཞུགས་པ། ཞབས་གཡས་བསྐུམ་གཡོན་བརྐྱང་ནས་མེ་རི་འབར་བའི་དབུས་ན་བཞུགས་པ། སྤྱི་བོ་ན་ཟླ་བའི་སྟེང་ན་འོད་དཔག་ཏུ་མེད་པས་དབུར་བརྒྱན་པར་བསམས་ལ་བསྟོད་པ་ཚར་གཅིག་གདོན། དེ་ནས་སྔ་མ་བཞིན་དུ་བསྟོད་པ་འདོན་ན་བྱ་གསུང༌།

གསང་སྔགས་ཟློས་ན་སྙིང་གར་པདྨ་འདབ་མ་བཅུ་གཉིས་པའི་ལྟེ་བར། ཨོཾ་དང་ཧཱའི་བར་དུ་མི་སྤེལ་ལ་འདབ་མ་རྣམས་ལ་ཡི་གེ་ལྷག་མ་བཀོད་པ་དེ་ལ་དམིགས་ལ། ཨོཾ་ན་མསྟཱ་རེ་ན་མོ་ཧ་ར་ཧཱུྃ་ཧ་ར་སྭཱ་ཧཱ། ཞེས་བཟླ་བར་བྱའོ། །དེ་ལྟར་བསྙེན་པའི་གྲངས་ནི། འབུམ་ཕྲག་བཅུ་བཟླས་ན། གནས་སྐབས་ཀྱི་འབྲས་བུ་ལས་ཐམས་ཅད་འགྲུབ་པ་དང༌། མཐར་ཐུག་སངས་རྒྱས་འགྲུབ་པར་འགྱུར་རོ། །ཉིན་མོ་ནི་ཞི་བའི་ང་རྒྱལ། མཚན་མོ་ནི་ཁྲོ་མོའི་ང་རྒྱལ་གྱིས་གནས་པར་བྱ་བས་སྤྱོད་ལམ་བྱའོ། །སྔགས་དེ་གཉིས་ཀྱང་བསྟོད་པ་རང་གི་གསེབ་ནས་བཏུས་པ་ཡིན་ནོ། །འདི་ནི་རྩ་བའི་སྔགས་ཀྱི་བསྟོད་པ་འདི་དང་ཞེས་པའི་དོན་ཡིན་ནོ། །ལྷ་དེ་གཉིས་ཀྱི་སྐུ་མདོག་ཕྱག་མཚན་ཡང་བསྟོད་པ་རང་གི་གསེབ་ནས་བྱུང་བ་ཡིན་གསུང༌། གཏོར་མ་གཏོང་བར་འདོད་ན་ཅི་འབྱོར་པའི་གཏོར་མ་མདུན་དུ་བཤམས་ལ། ཨོཾ་ཨཱཿཧཱུྃ་ལན་གསུམ་གྱིས་བདུད་རྩིར་བྱིན་གྱིས་བརླབས་ལ། མདུན་གྱི་ནམ་མཁར་རྗེ་བཙུན་མ་བསྐྱེད་ལ། དེ་ལ་ཡེ་ཤེས་པ་སྤྱན་དྲངས་ལ་བསྟིམ། དེའི་ལྗགས་རྡོ་རྗེའི་སྦུ་གུའི་འོད་ཟེར་གྱིས་གཏོར་མ་གསོལ་བར་བསམ་ཞིང་རང་རང་གི་སྔགས་གསུམ་མམ་བདུན་ནམ་ཉི་ཤུ་རྩ་གཅིག་བརྗོད་ལ་དབུལ། དེའི་རྗེས་ལ་པཱུ་ཛ་ལྔ་དབུལ། བསྟོད་པ་ཉི་ཤུ་རྩ་གཅིག་པས་བསྟོད། འདོད་པའི་དོན་ལ་གསོལ་བ་གདབ། སྤྱན་དྲངས་པ་གཤེགས་སུ་གསོལ། བསྐྱེད་པ་བདག་ལ་བསྡུ། གཏོར་མ་ས་གཙང་སར་གཏང་གསུང་ངོ༌། །ན་མ་སྟཱ་རཱ་ཡཻ། བརྟན་གནས་པའམ་གཞན་ཡང་འགྲོ་བ། ཟོས་པ་དང་ནི་འཐུངས་པ་ཉིད་ཀྱང༌། །ཞེས་པ་ལ་བརྟན་ནས་སེམས་ཅན་གནས་པ་དང་གཡོ་བའི་དུག་གིས་ཟིན་པ་ལ་ཕན་གདགས་པར་འདོད་ན་ཙཀྲ་གང་ཡང་རུང་བ་དེའི་སྤྱི་བོར་བསམ། དེ་ལས་བདུད་རྩིའི་ཆུ་རྒྱུན་བྱུང་བས་དུག་གི་ནད་དང་གདོན་ཐམས་ཅད་དྲི་མའི་ལམ་ནས་ཁ་དོག་ནག་པོའི་རྣམ་པས་འཐོན་ནས་བྱུང་བར་བསམ་ཞིང༌། ཞི་ཁྲོའི་སྔགས་གང་ཡང་རུང་བ་བཟླས་སོ། །རིམས་ལ་སོགས་པ་ནད་དྲག་པོ་གཞན་ལ་ཡང་སྦྱོར་གསུང་ངོ༌། །ན་མསྟཱ་རཱ་ཡཻ།

བུ་འདོད་པས་ནི་བུ་ཐོབ་འགྱུར་ཞིང༌། །ནོར་འདོད་པས་ནི་ནོར་རྣམས་ཉིད་ཐོབ། །ཅེས་པ་ལ་བརྟེན་ནས། བུ་དང་ནོར་འདོད་པས་བུམ་པ་མཚན་ཉིད་དང་ལྡན་པ་ལེགས་པར་བཅས། ནང་དུ་སྨན་ལྔ་འབྲུ་ལྔ་རིན་པོ་ཆེ་ལྔ་གཞུག ཁ་རྒྱན་དང་མགུལ་ཆིངས་བཅིངས། གཟུངས་ཐག་གདགས། རས་དཀར་པོའི་སྟེང་དུ་བཞག བུམ་པའི་མགྲིན་པའི་ནང་དུ་ཙཀྲ་གང་ཡང་རུང་བ་ཅིག་བསྒོམ། དེ་ལས་བདུད་རྩིའི་ཆུ་རྒྱུན་བྱུང་བས་བུམ་པ་གང་བར་བསམ་ཞིང༌། ཞི་ཁྲོ་གང་ཡང་རུང་བའི་སྔགས་བཟླས། བུམ་པའི་ཆུས་སྐྱེས་པའམ་བུད་མེད་དེའི་ལུས་བཀྲུ་ཁོང་དུ་ཡང་འཐུང་དུ་གཞུག ཡང་ཙཀྲ་དེ་གུར་གུམ་མམ། གི་ཧཾ་གིས་བྲིས་ལ་རབ་གནས་བྱས་ལ་སྐྱེས་པའི་དཔུང་པ་གཡས་ལ། བུད་མེད་ཀྱི་གཡོན་ལ། བྱིས་པའི་གང་ཡང་རུང་བའམ་མགུལ་དུ་གདགས་སོ། །བསྟོད་པ་ཉི་ཤུ་རྩ་གཅིག་པ་འདོན་དུ་གཞུག དངོས་གྲུབ་ཅི་འདོད་པ་རྗེ་བཙུན་མ་ལ་གསོལ་བ་གདབ། དེ་ལྟར་ཙཀྲ་དང་ཁྲུས་ཀྱི་ཆོ་ག་དེས་བུ་འདོད་པ་དང་ནོར་འདོད་པ་ཡང་འགྲུབ་པར་འགྱུར་རོ། །གཞན་ཡང་རིམས་དྲག་པོ་དང༌། ནད་དང༌། གནོད་དང༌། འབྱུང་པོའི་བར་ཆད་གང་ཡང་རུང་བ་ལ། ཆོ་ག་འདིས་ཕན་གདགས་པར་གསུངས་སོ། །ཤིན་ཏུ་བྱིན་རླབས་ཆེ་བའི་རྗེ་བཙུན་མའི་ལས་ཁ་ཚར་བ་རྫོགས་སོ། །འདིའི་བརྒྱུད་པ་ནི། ཇོ་བོ་ཏཻ་ལོ་པས། ནཱ་རོ་པ་ལ། དེས་བླ་མ་ཕམ་ཐིང་པ་ལ། དེས་མལ་གྱོ་ལོ་ཙཱ་བ་ལ། དེས་བླ་མ་ས་སྐྱ་པ་ལ། དེས་བླ་མ་བཀྲ་ཤིས་འོད་ལ་གནང་ངོ༌།། །།ཇོ་མོ་ལ་གསོལ་བ་འདེབས་པར་འདོད་ན། སྐུའི་དྲུང་དུ་མཆོད་པ་ཚར་གཅིག་དང༌། དགེ་སློང་བཞི་སྡེ་གཅིག་གིས་དང་པོ་སངས་རྒྱས་འདུས་སྐུ་ལ་སོགས་པ་ལ་ཕྱག་བྱས། མེ་ཏོག་དོར། མཆོད་པ་ཚར་གཅིག་དབུལ། ཡན་ལག་བདུན་པ་བསྡུས་པ་རེ་བྱ། དེ་ནས་ལྷ་དང་ལྷ་མིན་ཞེས་པས་ཕྱག་བདུན་བྱ། མེ་ཏོག་དོར། མཆོད་པ་འབུལ་བ་ལ་སོགས་པ་ཡན་ལག་བདུན་པ་བསྡུས་པ་རེ་བྱ། གོང་གི་དམ་ཚིག་པ་བསྐྱེད་པ་དང༌། ཡེ་ཤེས་པ་སྤྱན་དྲངས་ལ་བསྟོད་པ་ཚར་གཉིས་གདོན་པར་བྱའོ། །འདོད་པའི་དོན་ལ་གསོལ་བ་གདབ་བོ།

དེ་ནས་ཕྱག་ལ་སོགས་པ་སྔར་བཞིན་བྱས་ལ། བསྟོད་པ་ལན་གསུམ་གདོན་པར་བྱའོ། །ལན་གསུམ་པ་ལ་བསྟོད་པ་ཚར་བདུན་གདོན་པར་བྱའོ། །ཕྱག་འཚལ་ཞེས་པའི་སྐབས་སུ་ཡིད་ཀྱིས་ཕྱག་རེ་བྱའོ། །འདི་རྨི་ལམ་དང་མཚན་མ་ངན་པ་དང༌། ཐུགས་མ་བདེ་བ་ལ་སོགས་པ་ལ་བྱས་ན་བཟློག་པར་འགྱུར་རོ། །མ་འཚོགས་ན་རང་གཅིག་པུས་ཀྱང་བྱར་བཏུབ་བོ། །མཎྜལ་བཞི་པའི་མན་ངག་རྫོགས་སོ།། །།མངྒ་ལཾ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
