༼ན༽༄༅། །སྒྲ་སྒྲུབ་དུམ་བུ་དགུ་པ་བཞུགས།། ༄༅། །ནི་དཱ་ན། ཞེས་པའི་སྒྲ་ལས་དྲངས་ནས། ནིཤྩི་ཏ་ནི་ངེས་པ་དང༌། ཨུ་ད་ཡ་ནི་འབྱུང་བ་སྟེ། དེ་ལྟར་ན་དོན་གྱི་གླེང་གཞི་འབྱུང་བར་ངེས་པ་ལ་ཡང་འཇུག ཡང་ནིཤྩི་ཏ་ནི་ངེས་པ་དང༌། དཱ་ན་ནི་སྟེར་བ་སྟེ། དེ་ལྟར་ན་ཚིག་གི་གླེང་གཞི་ངེས་པ་སྟེར་བ་ལ་ཡང་འཇུག་གོ། །།ཨཱ་དི་ནི་གདོད་མའམ་དང་པོ། ཨ་ནུཏྤནྣ་སྐྱེ་བ་མེད་པའམ་མ་སྐྱེས་པ་དེ་དག་སྒྲ་པའི་ལུགས་ཀྱིས་སྦྱར་ན། ཨཱ་དི་ལས་ཨི་བཏོན་ལ། ད་སྲོག་མེད་དུ་བྱ། ཨི་ལས་ཨ་དང་ཡ་གཉིས་སུ་བྱ་སྟེ། དེའང་ཀ་ལཱ་པ་ལས། ཨི་ལས་ཨ་དང་ཡའོ། །ཞེས་འབྱུང་བའི་ཕྱིར་རོ། །དེའི་རྗེས་ལ་ཨ་ཡང་དབྱིའོ། །ཞེས་འབྱུང་བའི་མདོ་ལུས་ཀྱི། ཨ་ཡང་དབྱིའོ། །དེ་ནས་སྲོག་མེད་པའི་ཕྱིར་དང༌། ཡང་འདྲེའོ། །དེ་ནས་ཨ་ནུཏྤནྣ་ཨའི་ཨཱ་དྱ་ལ་སྲོག་ཏུ་ཐིམ་པས་དྱ་ནུཏྤནྣ་ཞེས་བྱ་བར་སྦྱར་རོ།། །།ལུས་ལ་དེ་ཧ་ཞེས་ཟེར་བ་དེ། དེ་ཧ་སཾ་བྷ་བ་ཏཱི་ཏྱ་སྨཱ་དེ་ཝ། ཞེས་པས་དང་པོའི་དེ་དང༌། ཐ་མའི་ཝ་སྦྱར་ནས་བར་གྱི་ཚིག་རྣམས་ཕྱིས་པས་དེ་ཝར་འགྱུར་ཏེ། སྒྲ་པའི་ལུགས་ཀྱིས་བར་གྱི་ཚིག་མི་མངོན་པའི་བསྡུ་བ་བྱ་བ་ཡིན། བོད་ཀྱི་དཔེ་ལ་སྦྱར་ན་གངས་ལས་བྱུང་བའི་ཆུ་ལ་གངས་ཆུ་ཞེས་ཟེར་བ་ལྟ་བུའོ། །ཡོ་ཥི་ཏ་ནི་མང་པོའི་སྒྲ་ལས་དྲངས་ན་ཡོ་ནི་ནི་སྐྱེ་གནས། ཥི་ཏའི་སྒྲ་ལས་དོན་གྱི་འབྱུང་བའམ་བསྐྱེད་པར་འགྲོ་བས་སྐྱེ་འགྲོ་བསྐྱེད་པའི་གནས། བྷ་གི་ནཱི་སྲིང་མོ། བྷ་གའི་སྒྲས་ནི། བཅོམ་པ་དང༌། སྐལ་བ་དང༌། ལེགས་པ་ལ་སོགས་པ་ལ་འཇུག་པས་འདིར་སྐལ་པ་ལ་སྦྱར། ཨི་ནི་སྒྲས་ལྡན་པའམ་སྟོན་པ་ལ་འཇུག དུ་ཧི་ཏཱ་ནི་བུ་མོ། །དུ་དུ་ཧ་ནཱ་འོ་མ་འཇོ་བ་ཡིན། ཉེད་པ་མརྡ་ན་ཟེར། མ་མརྡ་ནའི་སྒྲ་ལས་དྲངས་ནས། མཎྜལ་ལམ་མརྡ་ན་ཞེས་བརྗོད་ན་དཀྱིལ་འཁོར། སོར་མོ་ཉེད་པ་ལ་མརྡ་ན་ཟེར་བའི་སྒྲ་ལས་དྲངས་ནས། མུ་ཏྲཱ་ཕྱག་རྒྱ་ཡིན་པས། སོར་མོ་ཉེད་པའང་དེ་བཞིན་ཏེ་ཕྱག་རྒྱ་ཡིན་ནོ།།

།།ལེའུ་དང་པོ་འདིར་ཡང་སྡོམ་པ་རབ་དབྱེ་བཤད། ཕྱི་མའི་གསུམ་པ་དང༌། བཞི་པར་བཤད་པའི་ཟློས་པའི་སྐྱོན་མེད་དམ་སྙམ་ན། འདིར་རྒྱུ་ལ་སྦྱར་ཏེ་བཤད། ཕྱི་མའི་གསུམ་པར་དཀྱིལ་འཁོར་འཁོར་ལོ་ལ་སྦྱར་ཏེ་བཤད། ཕྱི་མའི་བཞི་པར་རང་བྱིན་གྱིས་བརླབས་པ་ལ་སྦྱར་ཏེ་བཤད་པས་ཟློས་པའི་སྐྱོན་མེད་དོ། །དུར་ཁྲོད་ཀྱི་སོལ་བ་ཞེས་པ་ལ་སོགས་པས་ནི། རྒྱུད་འདིར་ཡེ་ཤེས་ཀྱི་རིམ་པས་བྲི་བར་བསྟན་པ་ཡིན་ནོ། །དྲང་སྲོང་མ་ར་ག་ཞེས་པ་ནི་བདུད་རྒྱལ་ཏེ་དེའི་བུ་ཡིན་པས་ན་བདུད་ལས་རྒྱལ་བའི་བུའོ། །ཕྱིའི་དཀྱིལ་འཁོར་བསྒོམ་པ་སྔོན་དུ་སོང་ནས་ནང་ལུས་ཀྱི་དཀྱིལ་འཁོར་བསྒོམས་པའི་རྗེས་ལ་འཁོར་ལོ་གཉིས་ལ་བརྟེན་པའི་ཙཎྜཱ་ལཱི་བསྒོམ་པ་ནི་ཤིང་གཅིག་ཅེས་བྱ་བ་ལ་སོགས་པའོ། །དུམ་བུ་དགུ་པོ་འདི་ནི་ས་སྐྱ་པཎྜི་ཏ། ཨཱ་ནནྡ་དྷྭ་ཛ་ཤྲཱི་བྷ་དྲས་མཛད་པའོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
