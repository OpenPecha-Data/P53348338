༼བ༽༄༅། །སྤྱི་བོ་ལྷས་པ་བཤེས་གཉེན་རིན་ཆེན་ལ་བསྐུར་བ་བཞུགས།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། བླ་མ་དང་དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །གདོད་ནས་ཉེར་ཞི་ཞི་བའི་ལམ་གྱིས་ཞི་བ་དོན་གཉེར་སྐྱེ་བོ་རྣམས། །སྲིད་ཞིའི་ཆོས་གཉིས་ཉེ་བར་ཞི་བའི་ཞི་བདེ་མཆོག་ལ་འགོད་མཛད་པ། །ཉོན་མོངས་ཞི་ཞིང་དབང་པོ་ཉེར་ཞི་ཞི་བའི་འཁོར་གྱིས་ཡོངས་བསྐོར་བ། །ཞི་བའི་དཔལ་གྱིས་ཞི་ཉིད་རབ་འབྱོར་ཞི་བ་བླ་མེད་དེ་ལ་འདུད། །རྫོགས་པའི་སངས་རྒྱས་དཔལ་གྱི་ཕུལ། །རབ་མཆོག་དམ་པའི་གསུང་ལ་ཡིད། །ཀུན་ནས་དང་བའི་དཔལ་བརྙེས་པའི། །མི་ཡི་དབང་པོ་ཕྱོགས་བཅུར་རྒྱལ། །རྒྱ་མཚོའི་གོས་ཅན་རྒྱ་མཚོའི་མཐའ་གཏུགས་ས་ཆེན་འདི་ན་བློ་ཆེན་དུ། །རློམ་ཆེན་འགའ་ཡང་རློམ་པ་མི་མངའ་བློ་ཆེན་ཆེན་པོའི་མཁྱེན་རབ་ཀྱི། །ཆ་ཡི་ཆར་ཡང་ཉེ་བར་བསྒྲུན་མིན་ཆེས་ཆེར་གསལ་བའི་གྲགས་ལྡན་པ། །ཆོས་ཀྱི་རྒྱལ་པོ་བཟང་པོའི་ཞབས་ཀྱི་ཞབས་རྡུལ་གུས་ལེན་ཆོས་སྨྲ་བ། །བསོད་ནམས་རྒྱ་མཚོ་ཏིང་འཛིན་ཞི་བའི་པདྨོའི་ཚལ། །སྙན་པ་གྲགས་པའི་དྲི་བསུང་གིས་བསྡུས་སློབ་མ་ཡི། །བུང་བའི་སྤྲིན་ཕུང་གདམས་པའི་སྦྲང་རྩིས་ཚིམ་མཛད་པ། །བཤེས་གཉེན་རིན་ཆེན་པདྨོའི་ཚལ་ལ་དོན་འདི་ཞུ། །དེ་གཅིག་ཁོ་ནར་མ་ཟད་དེ། །དེ་ཡི་དབང་གིས་ཉེར་བསྡུས་པའི། །དགེ་འདུན་འདུས་པའི་རྒྱ་མཚོ་ལའང༌། །ཕྲིན་གྱི་ཚུལ་གྱིས་བསྙད་པར་བགྱི། །ཁྱེད་ཀྱི་ཐུགས་དགོངས་དག་པ་ལས་བྱུང་བའི་སྤྲིང་ཡིག བདེ་བར་གཤེགས་པའི་གསུང་རབ་རིན་པོ་ཆེའི་རྒྱུས་ཉེ་བར་བསྒྲུབས་པའི་རྒྱན་དང་བཅས་པ། ཡུལ་དང་ལྗོངས་དང༌། གྲོང་ཁྱེར་མང་པོའི་ལམ་ནས་བརྒྱུད་ནས་ཉེ་ཞོ་མེད་པར་ཁོ་བོའི་སུག་ཏུ་ཕྱིན་པས་ཡིད་དགའ་བ་ཡར་གྱི་ངོའི་ཟླ་བ་བཞིན་དུ་འཕེལ་བར་གྱུར། སླད་མ་སྐབས་ཀྱི་དོན་ལ། ཁྱེད་ཀྱིས་ཅི་སྐད་གསུངས་པ་བཞིན་དུ། སྔོན་གྱི་དམ་པ་རྣམས་ཀྱི་རྣམ་པར་ཐར་པ་དྲི་མ་མེད་ཅིང་དགོངས་པ་གཅིག་པ་ཡིན་ལ།

དེ་བཞིན་དུ་འོ་སྐོལ་རྣམ་པའང་ལྷག་པའི་བསམ་པ་འདྲ་བ་ཡིན་མོད། དེ་ལྟ་ན་ཡང༌། ཡུལ་གྱིས་ཆེར་བསྐལ་ཞིང༌། སྙིགས་མའི་དུས་ཀྱི་སྐྱེ་བོ་མང་པོས་ལོག་པར་བསྒྲུབས་པས་ཇི་སྐད་བསམས་པ་བཞིན་གྲུབ་པར་དཀའ་བ་ཁྱེད་ཀྱི་ཐུགས་ལ་གསལ། ངེད་སའི་ཕྱོགས་འདི་ཀུན་ཏུ་བསྡད་པའི་ཕྱོགས་ནས་ཀྱང་བསྟན་པ་དང་སེམས་ཅན་སྤྱི་ལ་ཕན་པ་འབད་ནས་བསྒྲུབས་པ་ཡིན་ཡང༌། མོས་པ་ཐ་དད་པའི་སེམས་ཅན་རྣམས་ཀྱིས་སྣ་ཚོགས་སུ་གླེང་བ་ཆོས་ཉིད་ཡིན། འོན་ཀྱང་བདེན་པའི་ངག་ལ་གཟུ་བོའི་ཚིག་གསན་ཏེ་མཆི། སླད་ནས་ཁྱེད་ཇི་སྐད་གསུང་བ་བཞིན་དུ་སྤྱིར་བསྟན་པ་དང༌། བྱེ་བྲག་ཁྱེད་ཀྱི་ཕྱོགས་སུའང་གང་ཕན་པ་དང༌། དོན་དུ་འགྱུར་བ་བྱེད་པ་ཡིན། ད་རེས་ཀྱི་སྐབས་སུ་མིའི་དབང་པོ་དང་ལྷན་ཅིག་ཏུ་ཡོད་ན། དངོས་སུ་བརྩོན་པ་བྱེད་པ་སྤྲོ་ནའང༌། ཅུང་ཟད་གནས་ཀྱིས་བསྐལ། འོན་ཀྱང་ཞུ་བའི་ཚིག་ནན་ཏན་ཅན་བསྐུར་བ་ལགས། ཕྱིན་ཆད་ངེད་ཀྱིས་ཀྱང་ཕན་ཚུན་ཇི་ལྟར་གླེངས་པ་བཞིན་དུ་ནུས་ཚད་ཀྱིས་བསྒྲུབ་པ་ཡིན། ཁྱེད་དཔོན་སློབ་རྣམས་ཀྱིས་ཀྱང་ཁྱེད་རང་ཇི་སྐད་གསུངས་པ་བཞིན་དུ་བསྒྲུབ་པ་ཁྱེད་རང་ལ་རག་ལས་པ་ཡིན། སངས་རྒྱས་ཉི་མ་གདུལ་བྱ་ཡི། །རི་ལ་ཉེ་བར་ནུབ་པར་གྱུར། །དེ་ཡི་བསྟན་པ་ཉི་འོད་ཀྱང༌། །ནུབ་ལ་ཉེ་བར་ཕྱོགས་པར་གྱུར། །དེ་ལྟར་དགོངས་ནས་བསྟན་པ་དང༌། །འགྲོ་ལ་གཅེས་པར་དགོངས་རྣམས་ཀྱིས། །དྲི་མེད་ལུང་དང་མི་འགལ་བར། །ཤིན་ཏུ་བརྩོན་པར་བྱ་བར་རིགས། །གསུང་རབ་རིན་ཆེན་གླིང་ལས་ལེགས་འོངས་ཤིང༌། །རྣམ་དཔྱོད་བློ་ཡིས་བྱི་དོར་ལེགས་བྱས་པ། །སྤྲིང་ཡིག་ལེགས་པར་བརྗོད་པའི་རིན་ཆེན་ཕྲེང༌། །བསམ་པ་དགེ་བས་ལེགས་པར་སྤྲིངས་པ་ཡིན། །དེ་ལས་དགེ་བ་དྲི་མེད་གང་ཐོབ་པས། །འགྲོ་ཀུན་དྲི་མེད་ཆོས་ཀྱི་ལམ་ཞུགས་ཏེ། །དྲི་མེད་བྱང་ཆུབ་སྤྱོད་པ་ཉེར་སྤྱད་ནས། །དྲི་མེད་ཆོས་ཀྱི་དབང་ཕྱུག་ཉིད་གྱུར་ཅིག །ཁྱི་ལོ་ཟླ་བ་བཅུ་པའི་ཚེས་བཅུ་གཅིག་ལ་འཕགས་པས་མིའི་དབང་པོའི་ཕོ་བྲང་ཁའི་ཕིང་ཧུར་བྲིས་པའོ། །ཡི་གེ་པ་ནི་ཨ་ཙ་རའོ།།

།། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
