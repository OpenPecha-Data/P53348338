༼བ༽༄༅། །ཐེག་ཆེན་སློབ་དཔོན་སློབ་མ་ལ་ལམ་ཉམས་སུ་བླང་ཚུལ་གྱི་རྣམ་དབྱེ་བསྟན་པ་བཞུགས།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། བླ་མ་དང་དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །གཙང་མཁས་རབ་བརྟན་བྱམས་པར་ལྡན། །ཡང་དག་ལམ་ལ་རྣམ་པར་གནས། །བདག་དང་གཞན་ལ་མ་ཆགས་པ། །གང་ཟག་དམ་པར་རྒྱལ་བས་བསྟན། །འདི་ནི་ཐེག་པ་ཆེན་པོ་དགེ་བའི་བཤེས་གཉེན་གྱི་མཚན་ཉིད་དང༌། སློབ་པ་པོ་གང་ཟག་གི་མཚན་ཉིད་དང༌། ལམ་ཉམས་སུ་བླང་བར་བྱ་བའི་མཚན་ཉིད་གསུམ་བསྟན་པར་རིགས་པར་བྱའོ། །དེ་དག་རིམ་པ་བཞིན་བསྟན་པར་བྱ་སྟེ། ཚུལ་གནས་ཐོས་དང་ཏིང་འཛིན་གྱིས། །ལུས་གཙང་སྨྲ་མཁས་སེམས་མི་གཡོ། །སེམས་ཅན་ཀུན་ལ་བྱམས་བརྩེར་ལྡན། །ཕ་རོལ་ཕྱིན་དྲུག་རྫོགས་པར་བྱེད། །བདག་མེད་ཡེ་ཤེས་ཚུལ་གྱིས་ནི། །གཟུང་དང་འཛིན་པར་མི་རྟོག་པ། །དེ་འདྲ་བཤེས་གཉེན་དམ་པར་འདོད། །ལུས་གཙང་ཀུན་ཏུ་སྤྱོད་ལ་མཁས། །དད་པའི་བསམ་པ་རབ་ཏུ་བརྟན། །ངང་གིས་གཞན་ལ་རྣམ་མི་འཚེ། །འཇིག་རྟེན་ཚུལ་ལུགས་ཇི་བཞིན་སྤྱོད། །ལུས་དང་ལོངས་སྤྱོད་ཆེད་ཆེར་མིན། །སློབ་པ་པ་ནི་དམ་པར་འདོད། །ལྷག་པའི་ཚུལ་ཁྲིམས་ཤེས་རབ་དང༌། །ཏིང་འཛིན་བསླབ་པ་རྣམ་གསུམ་གྱི། །ཉེས་པས་མི་གོས་ཐམས་ཅད་ཤེས། །མི་མཐུན་ཕྱོགས་ཀྱིས་རྣམ་མི་གཡོ། །བྱམས་བརྩེ་བྱང་སེམས་གཞི་བརྟན་ཞིང༌། །སྟོང་ཉིད་སྙིང་རྗེའི་སྙིང་པོ་ཅན། །ལམ་ལ་ཐབས་ལ་མཁས་པས་གནས། །བདག་ཉིད་ཐར་པར་མི་འདོད་ཅིང༌། །གནས་གཞན་མཚན་མར་འཛིན་ཆགས་བྲལ། །འདི་འདྲ་གང་གིས་ཤེས་དེ་མཁས། །ཤེས་པ་ཇི་བཞིན་སྤྱོད་པ་དེ། །རྫོགས་པའི་བྱང་ཆུབ་ཉེ་བ་ཡིན། །བློ་གྲོས་འཇིགས་མེད་རྒྱལ་མཚན་དཔལ། །བཟང་པོ་ཞེས་བྱས་ལེགས་པར་སྦྱར། །བྱང་ཆུབ་ཁོ་ནའི་དོན་དུ་བསྔོ། །ཤིང་མོ་ཡོས་བུའི་ལོ་རྒྱལ་གྱི་ཟླ་བའི་ཚེས་ཉི་ཤུ་ལྔའི་ཉིན་པར་དི་ཧིང་ཧུའི་དགོན་པར་སྦྱར་བའོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
