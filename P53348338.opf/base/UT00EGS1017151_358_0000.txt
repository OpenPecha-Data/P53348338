༼པ༽༄༅། །ཆོས་སྤྱོད་ཡན་ལག་བཅུ་པ་བཞུགས།། ༄༅། །ཨོཾ་སྭ་སྟི་སིདྡྷཾ། བླ་མ་དང་འཇམ་པའི་དབྱངས་ལ་ཕྱག་འཚལ་ལོ། །ཕྱོགས་དུས་ཀུན་ན་བཞུགས་པ་ཡི། །རྫོགས་སངས་རྒྱས་དང་དམ་པའི་ཆོས། །འཕགས་པའི་དགེ་འདུན་ཇི་སྙེད་པ། །དེ་སྙེད་ཀུན་ལ་ཕྱག་འཚལ་ལོ། །མེ་ཏོག་བདུག་སྤོས་མར་མེ་དང༌། །དྲི་དང་ཞལ་ཟས་མཆོད་པའི་སྤྲིན། །ནམ་མཁའི་མཐས་གཏུགས་ཇི་སྙེད་པ། །དེ་སྙེད་རྣམས་ཀྱིས་མཆོད་པར་བགྱི། །བདག་གིས་བྱང་ཆུབ་སྙིང་པོའི་བར། །སངས་རྒྱས་ཆོས་དང་ཚོགས་མཆོག་ནི། །ཇི་སྙེད་བཞུགས་པ་གང་ཡིན་པ། །དེ་སྙེད་རྣམས་ལ་སྐྱབས་སུ་མཆི། །ལས་དང་ཉོན་མོངས་དབང་གྱུར་པས། །ལུས་དང་ངག་དང་ཡིད་སྒོ་ནས། །སྡིག་པ་ཇི་སྙེད་བགྱིས་པ་ཀུན། །དེ་སྙེད་སོ་སོར་བཤགས་པར་བགྱི། །དུས་གསུམ་ཕྱོགས་བཅུའི་འཇིག་རྟེན་ན། །འཁོར་དང་མྱ་ངན་འདས་པ་ཡི། །དགེ་བའི་རྩ་བ་ཇི་སྙེད་པ། །དེ་སྙེད་ཀུན་ལ་ཡི་རང་ངོ༌། །སེམས་ཅན་ཀུན་གྱི་དོན་སླད་དུ། །འཁོར་ལོ་རིན་ཆེན་བསྐོར་བྱའི་ཕྱིར། །མཁའ་མཉམ་སངས་རྒྱས་ཇི་སྙེད་པ། །དེ་སྙེད་ཀུན་ལ་བསྐུལ་བར་བགྱི། །ཕན་པར་བཞེད་པ་དེ་དག་ཀུན། །མྱ་ངན་མི་འདའ་རྟག་བཞུགས་པར། །སེམས་ཅན་ཁམས་ནི་ཇི་སྙེད་པ། །དེ་སྙེད་དོན་དུ་གསོལ་བ་འདེབས། །རྒྱལ་བའི་རྗེས་སུ་བདག་ཞུགས་ཏེ། །བྱང་ཆུབ་མཆོག་ཏུ་སེམས་བསྐྱེད་པའི། །མཁའ་མཉམ་སེམས་ཅན་ཇི་སྙེད་པ། །དེ་སྙེད་དོན་ཀུན་བསྒྲུབ་པར་བགྱི། །དཔལ་ལྡན་ཡོན་ཏན་འབྱུང་བའི་གཞི། །སྟོང་པ་དང་ནི་སྙིང་རྗེའི་བདག །བྱང་ཆུབ་སྤྱོད་པ་ཇི་སྙེད་པ། །དེ་སྙེད་བདག་གིས་སྤྱད་པར་བགྱི། །བདག་དང་གཞན་གྱིས་དུས་གསུམ་དུ། །བསགས་པའི་ཟག་བཅས་ཟག་མེད་ཀྱི། །དགེ་བའི་རྩ་བ་ཇི་སྙེད་པ། །དེ་སྙེད་བྱང་ཆུབ་ཆེན་པོར་བསྔོ། །དཔལ་ལྡན་འཕགས་པས་འགྲོ་བ་ཀུན། །འཕགས་པའི་ས་ལ་དགོད་ཕྱིར་དང༌། །འཕགས་པའི་ཚོགས་ཀུན་མཉེས་བྱའི་ཕྱིར། །འཕགས་ལམ་ཡན་ལག་བཅུ་པ་སྦྱར། །ཆོས་སྤྱོད་ཡན་ལག་བཅུ་པ་ཞེས་བྱ་བ་རྫོགས་སོ།།

།། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
